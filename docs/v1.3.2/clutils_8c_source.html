301 CLUZone* clu_zone_new(cl_uint deviceType, cl_uint numQueues, cl_int queueProperties, clu_device_selector devSel, void* dsExtraArg, GError **err) {
302 
303  /* OpenCL status variable. */
304  cl_int status;
305 
306  /* OpenCL zone to initialize and return */
307  CLUZone* zone;
308 
309  /* Information about devices */
310  CLUDeviceInfo devInfos[CLU_MAX_DEVICES_TOTAL];
311 
312  /* Number of devices. */
313  cl_uint numDevices;
314 
315  /* Index of device information */
316  cl_int deviceInfoIndex;
317 
318  /* Context properties, */
319  cl_context_properties cps[3] = {CL_CONTEXT_PLATFORM, 0, 0};
320 
321  /* List of platform Ids. */
322  cl_platform_id platfIds[CLU_MAX_PLATFORMS];
323 
324  /* Number of platforms. */
325  cl_uint numPlatforms;
326 
327  /* Total number of devices. */
328  unsigned int totalNumDevices;
329 
330  /* Device IDs for a given platform. */
331  cl_device_id devIds[CLU_MAX_DEVICES_PER_PLATFORM];
332 
333  /* Initialize zone */
334  zone = (CLUZone*) malloc(sizeof(CLUZone));
335  gef_if_error_create_goto(
336  *err,
337  CLU_UTILS_ERROR,
338  NULL == zone,
339  CLU_ERROR_NOALLOC,
340  error_handler,
341  "Unable to allocate memory for OpenCL zone"
342  );
343  zone->context = NULL;
344  zone->queues = NULL;
345  zone->program = NULL;
346  zone->device_info.device_id = NULL;
347  zone->device_info.platform_id = NULL;
348  zone->device_info.device_name[0] = '\0';
349  zone->device_info.device_vendor[0] = '\0';
350  zone->device_info.platform_name[0] = '\0';
351 
352  /* Get number of platforms */
353  status = clGetPlatformIDs(0, NULL, &numPlatforms);
354  gef_if_error_create_goto(
355  *err,
356  CLU_UTILS_ERROR,
357  CL_SUCCESS != status,
358  CLU_OCL_ERROR,
359  error_handler,
360  "clu_zone_new: get number of platforms (OpenCL error %d: %s).",
361  status,
362  clerror_get(status));
363 
364  /* Get existing platforms */
365  status = clGetPlatformIDs(numPlatforms, platfIds, NULL);
366  gef_if_error_create_goto(
367  *err,
368  CLU_UTILS_ERROR,
369  CL_SUCCESS != status,
370  CLU_OCL_ERROR,
371  error_handler,
372  "clu_zone_new: get platform Ids (OpenCL error %d: %s).",
373  status,
374  clerror_get(status));
375 
376  /* Cycle through platforms, get specified devices in existing platforms */
377  totalNumDevices = 0;
378  for(unsigned int i = 0; i < numPlatforms; i++) {
379  /* Get specified devices for current platform */
380  status = clGetDeviceIDs(
381  platfIds[i],
382  deviceType,
383  CLU_MAX_DEVICES_PER_PLATFORM,
384  devIds,
385  &numDevices);
386  if (status != CL_DEVICE_NOT_FOUND) {
387  /* At least one device found, lets take note */
388  gef_if_error_create_goto(
389  *err,
390  CLU_UTILS_ERROR,
391  CL_SUCCESS != status,
392  CLU_OCL_ERROR,
393  error_handler,
394  "clu_zone_new: get device Ids (OpenCL error %d: %s).",
395  status,
396  clerror_get(status));
397  for (unsigned int j = 0; j < numDevices; j++) {
398  /* Keep device and platform IDs. */
399  devInfos[totalNumDevices].device_id = devIds[j];
400  devInfos[totalNumDevices].platform_id = platfIds[i];
401  /* Get device name. */
402  status = clGetDeviceInfo(
403  devIds[j],
404  CL_DEVICE_NAME,
405  sizeof(devInfos[totalNumDevices].device_name),
406  devInfos[totalNumDevices].device_name,
407  NULL);
408  gef_if_error_create_goto(
409  *err,
410  CLU_UTILS_ERROR,
411  CL_SUCCESS != status,
412  CLU_OCL_ERROR,
413  error_handler,
414  "clu_zone_new: get device name info (OpenCL error %d: %s).",
415  status,
416  clerror_get(status));
417  /* Get device vendor. */
418  status = clGetDeviceInfo(
419  devIds[j],
420  CL_DEVICE_VENDOR,
421  sizeof(devInfos[totalNumDevices].device_vendor),
422  devInfos[totalNumDevices].device_vendor,
423  NULL);
424  gef_if_error_create_goto(
425  *err,
426  CLU_UTILS_ERROR,
427  CL_SUCCESS != status,
428  CLU_OCL_ERROR,
429  error_handler,
430  "clu_zone_new: get device vendor info (OpenCL error %d: %s).",
431  status,
432  clerror_get(status));
433  /* Get platform name. */
434  status = clGetPlatformInfo(
435  platfIds[i],
436  CL_PLATFORM_VENDOR,
437  sizeof(devInfos[totalNumDevices].platform_name),
438  devInfos[totalNumDevices].platform_name,
439  NULL);
440  gef_if_error_create_goto(
441  *err,
442  CLU_UTILS_ERROR,
443  CL_SUCCESS != status,
444  CLU_OCL_ERROR,
445  error_handler,
446  "clu_zone_new: get platform info (OpenCL error %d: %s).",
447  status,
448  clerror_get(status));
449  /* Increment total number of found devices. */
450  totalNumDevices++;
451  }
452  }
453  }
454 
455  /* Check whether any devices of the specified type were found */
456  if (totalNumDevices == 0) {
457  /* No devices of the specified type where found, return with error. */
458  gef_if_error_create_goto(
459  *err,
460  CLU_UTILS_ERROR,
461  1,
462  CLU_ERROR_DEVICE_NOT_FOUND,
463  error_handler,
464  "clu_zone_new: device not found.");
465  } else {
466  /* Several compatible devices found, choose one with given selector function. */
467  deviceInfoIndex = devSel(devInfos, totalNumDevices, dsExtraArg);
468  /* Test return value of selector function (if it is out of range,
469  * there is a programming error). */
470  g_assert_cmpint(deviceInfoIndex, >=, -1);
471  g_assert_cmpint(deviceInfoIndex, <, totalNumDevices);
472  /* If selector function returned -1, then no device is selectable. */
473  if (deviceInfoIndex == -1) {
474  gef_if_error_create_goto(
475  *err,
476  CLU_UTILS_ERROR,
477  1,
478  CLU_ERROR_DEVICE_NOT_FOUND,
479  error_handler,
480  "clu_zone_new: specified device not found.");
481  }
482  }
483 
484  /* Store info about the selected device and platform. */
485  zone->device_type = deviceType;
486  zone->device_info = devInfos[deviceInfoIndex];
487 
488  /* Determine number of compute units for that device */
489  status = clGetDeviceInfo(
490  zone->device_info.device_id,
491  CL_DEVICE_MAX_COMPUTE_UNITS,
492  sizeof(cl_uint),
493  &zone->cu,
494  NULL);
495  gef_if_error_create_goto(
496  *err,
497  CLU_UTILS_ERROR,
498  CL_SUCCESS != status,
499  CLU_OCL_ERROR,
500  error_handler,
501  "clu_zone_new: get target device info (OpenCL error %d: %s).",
502  status,
503  clerror_get(status));
504 
505  /* Create a context on that device. */
506  cps[1] = (cl_context_properties) devInfos[deviceInfoIndex].platform_id;
507  zone->context = clCreateContext(cps, 1, &zone->device_info.device_id, NULL, NULL, &status);
508  gef_if_error_create_goto(
509  *err,
510  CLU_UTILS_ERROR,
511  CL_SUCCESS != status,
512  CLU_OCL_ERROR,
513  error_handler,
514  "clu_zone_new: creating context (OpenCL error %d: %s).",
515  status,
516  clerror_get(status));
517 
518  /* Create the specified command queues on that device */
519  zone->numQueues = numQueues;
520  zone->queues = (cl_command_queue*) malloc(numQueues * sizeof(cl_command_queue));
521  gef_if_error_create_goto(
522  *err,
523  CLU_UTILS_ERROR,
524  NULL == zone->queues,
525  CLU_ERROR_NOALLOC,
526  error_handler,
527  "Unable to allocate memory to keep OpenCL command queues in Zone."
528  );
529 
530  for (unsigned int i = 0; i < numQueues; i++) {
531  zone->queues[i] = clCreateCommandQueue(
532  zone->context,
533  zone->device_info.device_id,
534  queueProperties,
535  &status);
536  gef_if_error_create_goto(
537  *err,
538  CLU_UTILS_ERROR,
539  CL_SUCCESS != status,
540  CLU_OCL_ERROR,
541  error_handler,
542  "clu_zone_new: creating command queue (OpenCL error %d: %s).",
543  status,
544  clerror_get(status));
545  }
546 
547  /* If we got here, everything is OK. */
548  g_assert (err == NULL || *err == NULL);
549  goto finish;
550 
551 error_handler:
552  /* If we got here there was an error, verify that it is so. */
553  g_assert (err == NULL || *err != NULL);
554  /* Free OpenCL zone. */
555  if (zone != NULL) {
556  clu_zone_free(zone);
557  zone = NULL;
558  }
559 
560 finish:
561 
562  /* Return. */
563  return zone;
564 
565 }

579 int clu_program_create(CLUZone* zone, char** kernelFiles, cl_uint numKernelFiles, char* compilerOpts, GError **err) {
580 
581  /* Helper variables */
582  cl_int ocl_status, ocl_build_status;
583  int ret_status;
584  char * build_log = NULL;
585  size_t logsize;
586  char** source = NULL;
587 
588  /* Import kernels */
589  source = (char**) malloc(numKernelFiles * sizeof(char*));
590  gef_if_error_create_goto(
591  *err,
592  CLU_UTILS_ERROR,
593  NULL == source,
594  ret_status = CLU_ERROR_NOALLOC,
595  error_handler,
596  "Unable to allocate memory for kernels source file.");
597  for (unsigned int i = 0; i < numKernelFiles; i++) { source[i] = NULL; }
598  for (unsigned int i = 0; i < numKernelFiles; i++) {
599  source[i] = clu_source_load(kernelFiles[i], err);
600  gef_if_error_goto(*err, GEF_USE_GERROR, ret_status, error_handler);
601  }
602 
603  /* Load kernels sources and create program */
604  cl_program program = clCreateProgramWithSource(
605  zone->context,
606  numKernelFiles,
607  (const char**) source,
608  NULL,
609  &ocl_status);
610  gef_if_error_create_goto(
611  *err,
612  CLU_UTILS_ERROR,
613  CL_SUCCESS != ocl_status,
614  ret_status = CLU_OCL_ERROR,
615  error_handler,
616  "Create program with source (OpenCL error %d :%s).",
617  ocl_status,
618  clerror_get(ocl_status));
619 
620  /* Perform runtime source compilation of program */
621  ocl_build_status = clBuildProgram(
622  program,
623  1,
624  &zone->device_info.device_id,
625  compilerOpts,
626  NULL,
627  NULL);
628  /* Check for errors. */
629  if (ocl_build_status != CL_SUCCESS) {
630  /* If where here it's because program failed to build. However, error will only be thrown after getting build information. */
631  /* Get build log size. */
632  ocl_status = clGetProgramBuildInfo(
633  program,
634  zone->device_info.device_id,
635  CL_PROGRAM_BUILD_LOG,
636  0,
637  NULL,
638  &logsize);
639  gef_if_error_create_goto(
640  *err,
641  CLU_UTILS_ERROR,
642  CL_SUCCESS != ocl_status,
643  ret_status = CLU_OCL_ERROR,
644  error_handler,
645  "Error getting program build info (log size, OpenCL error %d: %s) after program failed to build (OpenCL error %d: %s).",
646  ocl_status,
647  clerror_get(ocl_status),
648  ocl_build_status,
649  clerror_get(ocl_build_status));
650  /* Alocate memory for build log. */
651  build_log = (char*) malloc(logsize);
652  gef_if_error_create_goto(
653  *err,
654  CLU_UTILS_ERROR,
655  NULL == build_log,
656  ret_status = CLU_ERROR_NOALLOC,
657  error_handler,
658  "Unable to allocate memory for build log after program failed to build with OpenCL error %d (%s).",
659  ocl_build_status,
660  clerror_get(ocl_build_status));
661  /* Get build log. */
662  ocl_status = clGetProgramBuildInfo(
663  program,
664  zone->device_info.device_id,
665  CL_PROGRAM_BUILD_LOG,
666  logsize,
667  build_log,
668  NULL);
669  gef_if_error_create_goto(
670  *err,
671  CLU_UTILS_ERROR,
672  CL_SUCCESS != ocl_status,
673  ret_status = CLU_OCL_ERROR,
674  error_handler,
675  "Error getting program build info (build log, OpenCL error %d: %s) after program failed to build (OpenCL error %d: %s).",
676  ocl_status,
677  clerror_get(ocl_status),
678  ocl_build_status,
679  clerror_get(ocl_build_status));
680  /* Throw error. */
681  gef_if_error_create_goto(
682  *err,
683  CLU_UTILS_ERROR,
684  1,
685  ret_status = CLU_OCL_ERROR,
686  error_handler,
687  "Failed to build program (OpenCL error %d: %s). \n\n **** Start of build log **** \n\n%s\n **** End of build log **** \n",
688  ocl_build_status,
689  clerror_get(ocl_build_status),
690  build_log);
691  }
692  zone->program = program;
693 
694  /* If we got here, everything is OK. */
695  g_assert (err == NULL || *err == NULL);
696  ret_status = CLU_SUCCESS;
697  goto finish;
698 
699 error_handler:
700  /* If we got here there was an error, verify that it is so. */
701  g_assert (err == NULL || *err != NULL);
702 
703 finish:
704 
705  /* Free stuff. */

875 cl_uint clu_info_device_selector(CLUDeviceInfo* devInfos, cl_uint numDevices, void* extraArg) {
876 
877  /* numDevices must be greater than 0. */
878  g_assert_cmpuint(numDevices, >, 0);
879 
880  /* Index of selected device. */
881  cl_int index = -1;
882  /* Pointer to CLUDeviceInfo structure. */
883  CLUDeviceInfo* info;
884  /* Number of devices found which conform to the information. */
885  int numValidDevs = 0;
886  /* If more than 1 device is found this aux. struct. will be passed
887  * to a user query function. */
888  CLUDeviceInfo validDevInfos[CLU_MAX_DEVICES_TOTAL];
889  /* Maps the aux. struct. dev. index to the main struct. dev. index. */
890  int map[CLU_MAX_DEVICES_TOTAL];
891  /* Flag to check if a device is conformant with the information. */
892  gboolean validDev;
893  /* Partial name must be a substring of complete name. */
894  gchar *partialName, *completeName;
895 
896  /* Check if extraArg contains a valid NULL-terminated array of strings. */
897  if (extraArg != NULL) {
898  info = (CLUDeviceInfo*) extraArg;
899  g_assert(info != NULL);
900  /* Cycle through available devices. */
901  for (unsigned int i = 0; i < numDevices; i++) {
902  /* Check for the three types of information: device name.
903  * device vendor and platform name. */
904  for (unsigned int j = 0; j < 3; j++) {
905  /* Not known if device will pass this filter. */
906  validDev = FALSE;
907  /* Obtain complete and partial information for comparison. */
908  switch (j) {
909  case 0:
910  if (info->device_name != NULL) {
911  completeName = g_ascii_strdown(devInfos[i].device_name, -1);
912  partialName = g_ascii_strdown(info->device_name, -1);
913  }
914  break;
915  case 1:
916  if (info->device_vendor != NULL) {
917  completeName = g_ascii_strdown(devInfos[i].device_vendor, -1);
918  partialName = g_ascii_strdown(info->device_vendor, -1);
919  }
920  break;
921  case 2:
922  if (info->platform_name != NULL) {
923  completeName = g_ascii_strdown(devInfos[i].platform_name, -1);
924  partialName = g_ascii_strdown(info->platform_name, -1);
925  }
926  break;
927  default:
928  g_assert_not_reached();
929  }
930  /* Check if partial name is within in the complete name. */
931  if (g_strrstr(completeName, partialName)) {
932  /* Valid device so far. */
933  validDev = TRUE;
934  }
935  /* Free temporary strings. */
936  g_free(partialName);
937  g_free(completeName);
938  /* If device didn't conform to info, go to next device. */
939  if (!validDev) {
940  break;
941  }
942  }
943  /* If device passed filters, add it to valid devices. */
944  if (validDev) {
945  validDevInfos[numValidDevs] = devInfos[i];
946  map[numValidDevs] = i;
947  numValidDevs++;
948  }
949  }
950  }
951 
952  /* Check result. */
953  if (numValidDevs == 1) {
954  /* Only one valid device found, return that one. */
955  index = map[0];
956  } else if (numValidDevs > 1) {
957  /* Several valid devices found, ask the user to chose one. */
958  index = map[clu_menu_device_selector_query(validDevInfos, numValidDevs)];
959  }
960 
961  /* Return device index. */
962  return index;
963 }
