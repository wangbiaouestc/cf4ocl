45 ProfCLProfile* profcl_profile_new() {
46 
47  /* Allocate memory for new profile data structure. */
48  ProfCLProfile* profile = (ProfCLProfile*) malloc(sizeof(ProfCLProfile));
49 
50  /* If allocation successful... */
51  if (profile != NULL) {
52  /* ... create table of unique events, ... */
53  profile->unique_events = g_hash_table_new(
54  g_str_hash,

72  /* ... and set timer structure to NULL. */
73  profile->timer = NULL;
74  /* ... and set total times to 0. */
75  profile->totalEventsTime = 0;
76  profile->totalEventsEffTime = 0;
77  /* ... and set absolute start time to maximum possible. */
78  profile->startTime = CL_ULONG_MAX;
79  }

164

166  cl_int ocl_status;
167  /* Function return status. */
168  int ret_status;
169  /* Unique event ID. */

185

205  &q1,
206  NULL);
207  gef_if_error_create_goto(
208  *err,

211  ret_status = PROFCL_OCL_ERROR,
212  error_handler,

309  /* Check if command queue of start event is already registered in the command queues table... */

389 gint profcl_evinst_comp(gconstpointer a, gconstpointer b, gpointer userdata) {
390  /* Cast input parameters to event instant data structures. */
391  ProfCLEvInst* evInst1 = (ProfCLEvInst*) a;
392  ProfCLEvInst* evInst2 = (ProfCLEvInst*) b;
393  ProfCLEvSort* sortType = (ProfCLEvSort*) userdata;
394  /* Perform comparison. */
395  if (*sortType == PROFCL_EV_SORT_INSTANT) {
396  /* Sort by instant */
397  if (evInst1->instant > evInst2->instant) return 1;
398  if (evInst1->instant < evInst2->instant) return -1;
399  } else if (*sortType == PROFCL_EV_SORT_ID) {
400  /* Sort by ID */
401  if (evInst1->id > evInst2->id) return 1;
402  if (evInst1->id < evInst2->id) return -1;
403  if (evInst1->type == PROFCL_EV_END) return 1;
404  if (evInst1->type == PROFCL_EV_START) return -1;
405  }
406  return 0;
407 }

420 int profcl_profile_overmat(ProfCLProfile* profile, GError** err) {
421 
422  /* Return status. */
423  int status;
424  /* Total overlap time. */
425  cl_ulong totalOverlap = 0;
426  /* Overlap matrix. */
427  cl_ulong* overlapMatrix = NULL;
428  /* Number of unique events. */
429  guint numUniqEvts;
430  /* Helper table to account for all overlapping events. */
431  GHashTable* overlaps = NULL;
432  /* Occurring events table. */
433  GHashTable* eventsOccurring = NULL;
434  /* Type of sorting to perform. */
435  ProfCLEvSort sortType;
436  /* Container for current event instants. */
437  GList* currEvInstContainer;
438 
439  /* Make sure profile is not NULL. */
440  g_assert(profile != NULL);
441 
442  /* Determine number of unique events. */
443  numUniqEvts = g_hash_table_size(profile->unique_events);
444 
445  /* Initialize overlap matrix. */
446  overlapMatrix =
447  (cl_ulong*) malloc(numUniqEvts * numUniqEvts * sizeof(cl_ulong));
448  gef_if_error_create_goto(
449  *err,
450  PROFCL_ERROR,
451  overlapMatrix == NULL,
452  status = PROFCL_ALLOC_ERROR,
453  error_handler,
454  "Unable to allocate memory for overlapMatrix.");
455  for (guint i = 0; i < numUniqEvts * numUniqEvts; i++)
456  overlapMatrix[i] = 0;
457 
458  /* Initialize helper table to account for all overlapping events. */
459  overlaps = g_hash_table_new(g_direct_hash, g_direct_equal);
460  gef_if_error_create_goto(
461  *err,
462  PROFCL_ERROR,
463  overlaps == NULL,
464  status = PROFCL_ALLOC_ERROR,
465  error_handler,
466  "Unable to allocate memory for overlaps helper table: g_hash_table_new function returned NULL.");
467 
468  /* Setup ocurring events table (key: eventID, value: uniqueEventID) */
469  eventsOccurring = g_hash_table_new(g_int_hash, g_int_equal);
470  gef_if_error_create_goto(
471  *err, PROFCL_ERROR,
472  eventsOccurring == NULL,
473  status = PROFCL_ALLOC_ERROR,
474  error_handler,
475  "Unable to allocate memory for occurring events table: g_hash_table_new function returned NULL.");
476 
477  /* Sort all event instants. */
478  sortType = PROFCL_EV_SORT_INSTANT;
479  profile->event_instants = g_list_sort_with_data(
480  profile->event_instants,
481  profcl_evinst_comp,
482  (gpointer) &sortType);
483 
484  /* Iterate through all event instants */
485  currEvInstContainer = profile->event_instants;
486  while (currEvInstContainer) {
487 
488  /* Loop aux. variables. */
489  ProfCLEvInst* currEvInst = NULL; /* Current event instant. */
490  GHashTable* innerTable = NULL; /* Inner hash table (is value for overlap hash table). */
491  GHashTableIter iter; /* Hash table iterator. */
492  gpointer key_eid, ueid_curr_ev, ueid_occu_ev; /* Hashtable key, unique event ID for current event, unique event ID for occurring event. */
493  guint eid_key1, eid_key2; /* Keys for start and end event instants. */
494  cl_ulong effOverlap; /* Event overlap in nanoseconds. */
495 
496  /* Get current event instant. */
497  currEvInst = (ProfCLEvInst*) currEvInstContainer->data;
498 
499  /* Check if event time is START or END time */
500  if (currEvInst->type == PROFCL_EV_START) {
501  /* Event START instant. */
502 
503  /* 1 - Check for overlaps with ocurring events */
504 
505  g_hash_table_iter_init(&iter, eventsOccurring);
506  while (g_hash_table_iter_next (&iter, &key_eid, NULL)) {
507 
508  /* The first hash table key will be the smaller event id. */
509  eid_key1 = currEvInst->id <= *((guint*) key_eid)
510  ? currEvInst->id
511  : *((guint*) key_eid);
512  /* The second hash table key will be the larger event id. */
513  eid_key2 = currEvInst->id > *((guint*) key_eid)
514  ? currEvInst->id
515  : *((guint*) key_eid);
516  /* Check if the first key (smaller id) is already in the hash table... */
517  if (!g_hash_table_lookup_extended(overlaps,
518  GUINT_TO_POINTER(eid_key1), NULL,
519  (gpointer) &innerTable)) {
520  /* ...if not in table, add it to table, creating a new
521  * inner table as value. Inner table will be initalized
522  * with second key (larger id) as key and event start
523  * instant as value. */
524  innerTable = g_hash_table_new_full(
525  g_direct_hash,
526  g_direct_equal,
527  NULL,
528  (GDestroyNotify) g_hash_table_destroy);
529  gef_if_error_create_goto(
530  *err,
531  PROFCL_ERROR,
532  innerTable == NULL,
533  status = PROFCL_ALLOC_ERROR,
534  error_handler,
535  "Unable to allocate memory for events inner table: g_hash_table_new_full function returned NULL.");
536  g_hash_table_insert(
537  overlaps, GUINT_TO_POINTER(eid_key1), innerTable);
538  }
539  /* Add second key (larger id) to inner tabler, setting the
540  * start instant as the value. */
541  g_hash_table_insert(
542  innerTable,
543  GUINT_TO_POINTER(eid_key2),
544  &(currEvInst->instant));
545  }
546 
547  /* 2 - Add event to occurring events. */
548  g_hash_table_insert(
549  eventsOccurring,
550  &(currEvInst->id), /* eid */
551  g_hash_table_lookup(profile->unique_events, currEvInst->eventName) /* ueid */
552  );
553 
554  } else {
555  /* Event END instant. */
556 
557  /* 1 - Remove event from ocurring events */
558  g_hash_table_remove(eventsOccurring, &(currEvInst->id));
559 
560  /* 2 - Check for overlap termination with current events */
561  g_hash_table_iter_init(&iter, eventsOccurring);
562  while (g_hash_table_iter_next(&iter, &key_eid, &ueid_occu_ev)) {
563  /* The first hash table key will be the smaller event id. */
564  eid_key1 = currEvInst->id <= *((guint*) key_eid)
565  ? currEvInst->id
566  : *((guint*) key_eid);
567  /* The second hash table key will be the larger event id. */
568  eid_key2 = currEvInst->id > *((guint*) key_eid)
569  ? currEvInst->id
570  : *((guint*) key_eid);
571  /* Get effective overlap in nanoseconds. */
572  innerTable = g_hash_table_lookup(
573  overlaps, GUINT_TO_POINTER(eid_key1));
574  effOverlap =
575  currEvInst->instant
576  -
577  *((cl_ulong*) g_hash_table_lookup(
578  innerTable, GUINT_TO_POINTER(eid_key2)));
579  /* Add overlap to overlap matrix. */
580  ueid_curr_ev = g_hash_table_lookup(
581  profile->unique_events, currEvInst->eventName);
582  guint ueid_min = GPOINTER_TO_UINT(ueid_curr_ev) <= GPOINTER_TO_UINT(ueid_occu_ev)
583  ? GPOINTER_TO_UINT(ueid_curr_ev)
584  : GPOINTER_TO_UINT(ueid_occu_ev);
585  guint ueid_max = GPOINTER_TO_UINT(ueid_curr_ev) > GPOINTER_TO_UINT(ueid_occu_ev)
586  ? GPOINTER_TO_UINT(ueid_curr_ev)
587  : GPOINTER_TO_UINT(ueid_occu_ev);
588  overlapMatrix[ueid_min * numUniqEvts + ueid_max] += effOverlap;
589  totalOverlap += effOverlap;
590  }
591  }
592 
593  /* Get next event instant. */
594  currEvInstContainer = currEvInstContainer->next;
595  }
596 
597  /* Add a pointer to overlap matrix to the profile. */
598  profile->overmat = overlapMatrix;
599 
600  /* Determine and save effective events time. */
601  profile->totalEventsEffTime = profile->totalEventsTime - totalOverlap;
602 
603  /* If we got here, everything is OK. */
604  g_assert (err == NULL || *err == NULL);
605  status = PROFCL_SUCCESS;
606  goto finish;
607 
608 error_handler:
609  /* If we got here there was an error, verify that it is so. */
610  g_assert (err == NULL || *err != NULL);
611 
612 finish:
613 
614  /* Free overlaps hash table. */
615  if (overlaps) g_hash_table_destroy(overlaps);
616 
617  /* Free eventsOccurring hash table. */
618  if (eventsOccurring) g_hash_table_destroy(eventsOccurring);
619 
620  /* Return status. */
621  return status;
622 }

633 int profcl_profile_aggregate(ProfCLProfile* profile, GError** err) {
634 
635  /* Return status. */
636  int status;
637  /* Hash table iterator. */
638  GHashTableIter iter;
639  /* A pointer for a event name. */
640  gpointer eventName;
641  /* Aggregate event info. */
642  ProfCLEvAggregate* evagg = NULL;
643  /* Type of sorting to perform on event list. */
644  ProfCLEvSort sortType;
645  /* Aux. pointer for event data structure kept in a GList. */
646  GList* currEvInstContainer = NULL;
647  /* A pointer to a ProfCLEvAggregate (agg. event info) variable. */
648  gpointer valueAgg;
649  /* Auxiliary aggregate event info variable.*/
650  ProfCLEvAggregate* currAgg = NULL;
651 
652  /* Make sure profile is not NULL. */
653  g_assert(profile != NULL);
654 
655  /* Initalize table, and set aggregate values to zero. */
656  g_hash_table_iter_init(&iter, profile->unique_events);
657  while (g_hash_table_iter_next(&iter, &eventName, NULL)) {
658  evagg = profcl_aggregate_new(eventName);
659  gef_if_error_create_goto(
660  *err,
661  PROFCL_ERROR,
662  evagg == NULL,
663  status = PROFCL_ALLOC_ERROR,
664  error_handler,
665  "Unable to allocate memory for ProfCLEvAggregate object.");
666  evagg->totalTime = 0;
667  g_hash_table_insert(profile->aggregate, eventName, (gpointer) evagg);
668  }
669 
670  /* Sort event instants by eid, and then by START, END order. */
671  sortType = PROFCL_EV_SORT_ID;
672  profile->event_instants = g_list_sort_with_data(
673  profile->event_instants, profcl_evinst_comp, (gpointer) &sortType);
674 
675  /* Iterate through all event instants and determine total times. */
676  currEvInstContainer = profile->event_instants;
677  while (currEvInstContainer) {
678 
679  /* Loop aux. variables. */
680  ProfCLEvInst* currEvInst = NULL;
681  cl_ulong startInst, endInst;
682 
683  /* Get START event instant. */
684  currEvInst = (ProfCLEvInst*) currEvInstContainer->data;
685  startInst = currEvInst->instant;
686 
687  /* Get END event instant */
688  currEvInstContainer = currEvInstContainer->next;
689  currEvInst = (ProfCLEvInst*) currEvInstContainer->data;
690  endInst = currEvInst->instant;
691 
692  /* Add new interval to respective aggregate value. */
693  currAgg = (ProfCLEvAggregate*) g_hash_table_lookup(
694  profile->aggregate, currEvInst->eventName);
695  currAgg->totalTime += endInst - startInst;
696  profile->totalEventsTime += endInst - startInst;
697 
698  /* Get next START event instant. */
699  currEvInstContainer = currEvInstContainer->next;
700  }
701 
702  /* Determine relative times. */
703  g_hash_table_iter_init(&iter, profile->aggregate);
704  while (g_hash_table_iter_next(&iter, &eventName, &valueAgg)) {
705  currAgg = (ProfCLEvAggregate*) valueAgg;
706  currAgg->relativeTime =
707  ((double) currAgg->totalTime)
708  /
709  ((double) profile->totalEventsTime);
710  }
711 
712  /* If we got here, everything is OK. */
713  g_assert (err == NULL || *err == NULL);
714  status = PROFCL_SUCCESS;
715  goto finish;
716 
717 error_handler:
718  /* If we got here there was an error, verify that it is so. */
719  g_assert (err == NULL || *err != NULL);
720

1014  profile->event_instants, profcl_evinst_comp, (gpointer) &sortType);
