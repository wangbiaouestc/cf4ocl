34 static void create_info_destroy_test() {
35 
36  /* Test variables. */
37  CCLContext* ctx = NULL;
38  CCLContext* ctx_aux = NULL;
39  cl_context context = NULL;
40  CCLDevice* dev = NULL;
41  CCLDevice* dev_aux = NULL;
42  cl_device_id device = NULL;
43  CCLQueue* cq = NULL;
44  cl_command_queue command_queue = NULL;
45  GError* err = NULL;
46  cl_command_queue_properties prop;
47  const cl_queue_properties prop_full[] =
48  { CL_QUEUE_PROPERTIES, CL_QUEUE_PROFILING_ENABLE, 0 };
49  cl_int ocl_status;
50 
51  /* Get the test context with the pre-defined device. */
52  ctx = ccl_test_context_new(&err);
53  g_assert_no_error(err);
54 
55  /* Get first device in context. */
56  dev = ccl_context_get_device(ctx, 0, &err);
57  g_assert_no_error(err);
58 
59  /* Test three ways to create a queue. */
60  for (cl_uint i = 0; i < 3; ++i) {
61 
62  /* Create command queue wrapper. */
63  switch (i) {
64  case 0:
65  /* The regular way. */
66  cq = ccl_queue_new(ctx, dev, CL_QUEUE_PROFILING_ENABLE,
67  &err);
68  g_assert_no_error(err);
69  break;
70  case 1:
71  /* Using the "full" constructor. */
72  cq = ccl_queue_new_full(ctx, dev, prop_full, &err);
73  g_assert_no_error(err);
74  break;
75  case 2:
76  /* Using the "wrap" constructor. */
77  CCL_BEGIN_IGNORE_DEPRECATIONS
78  command_queue = clCreateCommandQueue(
79  ccl_context_unwrap(ctx), ccl_device_unwrap(dev),
80  CL_QUEUE_PROFILING_ENABLE, &ocl_status);
81  g_assert_cmpint(ocl_status, ==, CL_SUCCESS);
82  CCL_END_IGNORE_DEPRECATIONS
83  cq = ccl_queue_new_wrap(command_queue);
84  g_assert_cmphex(GPOINTER_TO_UINT(command_queue), ==,
85  GPOINTER_TO_UINT(ccl_queue_unwrap(cq)));
86  break;
87  }
88 
89  /* Get some info and check if the return value is as expected. */
90  context = ccl_queue_get_info_scalar(
91  cq, CL_QUEUE_CONTEXT, cl_context, &err);
92  g_assert_no_error(err);
93  g_assert_cmphex(
94  GPOINTER_TO_UINT(context),
95  ==,
96  GPOINTER_TO_UINT(ccl_context_unwrap(ctx)));
97 
98  ctx_aux = ccl_queue_get_context(cq, &err);
99  g_assert_no_error(err);
100  g_assert_cmphex(
101  GPOINTER_TO_UINT(ctx), ==, GPOINTER_TO_UINT(ctx_aux));
102 
103  device = ccl_queue_get_info_scalar(
104  cq, CL_QUEUE_DEVICE, cl_device_id, &err);
105  g_assert_no_error(err);
106  g_assert_cmphex(
107  GPOINTER_TO_UINT(device),
108  ==,
109  GPOINTER_TO_UINT(ccl_device_unwrap(dev)));
110 
111  dev_aux = ccl_queue_get_device(cq, &err);
112  g_assert_no_error(err);
113  g_assert_cmphex(
114  GPOINTER_TO_UINT(dev), ==, GPOINTER_TO_UINT(dev_aux));
115 
116  prop = ccl_queue_get_info_scalar(
117  cq, CL_QUEUE_PROPERTIES, cl_command_queue_properties, &err);
118  g_assert_no_error(err);
119  g_assert_cmphex(prop, ==, CL_QUEUE_PROFILING_ENABLE);
120 
121  /* Destroy queue. */
122  ccl_queue_destroy(cq);
123  cq = NULL;
124  }
125 
126  /* Destroy context. */
127  ccl_context_destroy(ctx);
128 
129  /* Confirm that memory allocated by wrappers has been properly
130  * freed. */
131  g_assert(ccl_wrapper_memcheck());
132 
133 }

138 static void ref_unref_test() {

140  /* Test variables. */
141  CCLContext* ctx = NULL;
142  CCLQueue* cq = NULL;
143  CCLDevice* dev = NULL;
144  GError* err = NULL;
145 
146  /* Get the test context with the pre-defined device. */

153 
154  /* Create command queue wrapper. */
155  cq = ccl_queue_new(ctx, dev, 0, &err);

157

159  ccl_queue_ref(cq);

163

168  g_assert_cmpuint(1, ==, ccl_wrapper_ref_count((CCLWrapper*) cq));
169 
170  /* Destroy stuff. */
171  ccl_queue_unref(cq);
172  ccl_context_destroy(ctx);
173 
174  /* Confirm that memory allocated by wrappers has been properly
175  * freed. */
176  g_assert(ccl_wrapper_memcheck());
177 }

183 static void barrier_marker_test() {
184 
185  /* Test variables. */
186  CCLContext* ctx = NULL;
187  CCLDevice* dev = NULL;
188  CCLQueue* cq = NULL;
189  CCLBuffer* buf = NULL;
190  CCLEvent* evt_cq = NULL;
191  CCLEvent* evt_map = NULL;
192  CCLEvent* evt_unmap = NULL;
193  CCLEvent* evt_barrier = NULL;
194  CCLEvent* evt_marker = NULL;
195  GError* err = NULL;
196  cl_ulong* host_buf;
197  CCLEventWaitList ewl = NULL;
198  cl_int exec_status = -1;
199  cl_command_type ct = 0;
200  cl_uint ocl_ver;
201  cl_bool evt_in_cq;
202 
203  /* Get the test context with the pre-defined device. */
204  ctx = ccl_test_context_new(&err);
205  g_assert_no_error(err);
206 
207  /* Get first device in context. */
208  dev = ccl_context_get_device(ctx, 0, &err);
209  g_assert_no_error(err);
210 
211  /* Get OpenCL version for context. */
212  ocl_ver = ccl_context_get_opencl_version(ctx, &err);
213  g_assert_no_error(err);
214 
215  /* Create a command queue. */
216  cq = ccl_queue_new(ctx, dev, 0, &err);
217  g_assert_no_error(err);
218 
219  /* Create a device buffer. */
220  buf = ccl_buffer_new(
221  ctx, CL_MEM_READ_WRITE, 8 * sizeof(cl_ulong), NULL, &err);
222  g_assert_no_error(err);
223 
224  /* Map device buffer, get an event and analise it. */
225  host_buf = ccl_buffer_enqueue_map(buf, cq, CL_FALSE, CL_MAP_WRITE,
226  0, 8 * sizeof(cl_ulong), NULL, &evt_map, &err);
227  g_assert_no_error(err);
228 
229  /* Enqueue a barrier for waiting on the map event. */
230  evt_barrier =
231  ccl_enqueue_barrier(cq, ccl_ewl(&ewl, evt_map, NULL), &err);
232  g_assert_no_error(err);
233 
234  /* Check that the barrier event is CL_COMMAND_BARRIER or
235  * CL_COMMAND_MARKER (this depends on the OpenCL version, as well
236  * as the specific implementation). */
237  ct = ccl_event_get_command_type(evt_barrier, &err);
238  g_assert_no_error(err);
239  g_assert((ct == CL_COMMAND_BARRIER) || (ct == CL_COMMAND_MARKER));
240 
241  /* Unmap buffer, get resulting event. */
242  evt_unmap = ccl_buffer_enqueue_unmap(buf, cq, host_buf, NULL, &err);
243  g_assert_no_error(err);
244 
245 #ifdef CL_VERSION_1_2
246  if (ocl_ver >= 120) {
247  /* For OpenCL >= 1.2 the marker event is complete when either
248  * a) all previous enqueued events have completed, or b) the
249  * events specified in the event wait list have completed. */
250  ccl_event_wait_list_add(
251  &ewl, evt_map, evt_barrier, evt_unmap, NULL);
252  evt_marker = ccl_enqueue_marker(cq, &ewl, &err);
253  g_assert_no_error(err);
254  } else {
255 #endif
256  /* For OpenCL <= 1.1, the marker event is complete ONLY when all
257  * previous enqueued events have completed. */
258  CCL_UNUSED(ocl_ver);
259  evt_marker = ccl_enqueue_marker(cq, NULL, &err);
260  g_assert_no_error(err);
261 #ifdef CL_VERSION_1_2
262  }
263 #endif
264 
265  /* Wait for marker event to complete (which means all previous
266  * enqueued events have also completed). */
267  ccl_event_wait(ccl_ewl(&ewl, evt_marker, NULL), &err);
268 
269  /* Test queue flush. */
270  ccl_queue_flush(cq, &err);
271  g_assert_no_error(err);
272 
273  /* Check that the unmap event is CL_COMPLETE. */
274  exec_status = ccl_event_get_info_scalar(
275  evt_unmap, CL_EVENT_COMMAND_EXECUTION_STATUS, cl_int, &err);
276  g_assert_no_error(err);
277  g_assert_cmpint(exec_status, ==, CL_COMPLETE);
278 
279  /* Check that the event is a marker event. */
280  ct = ccl_event_get_command_type(evt_marker, &err);
281  g_assert_no_error(err);
282  g_assert_cmpuint(ct, ==, CL_COMMAND_MARKER);
283 
284  /* Check that generated events are in queue. */
285  CCLEvent* evts[] =
286  {evt_map, evt_barrier, evt_unmap, evt_marker, NULL};
287  for (cl_uint i = 0; evts[i] != NULL; ++i) {
288  ccl_queue_iter_event_init(cq);
289  evt_in_cq = CL_FALSE;
290  while ((evt_cq = ccl_queue_iter_event_next(cq)) != NULL) {
291  if (evt_cq == evt_map) {
292  evt_in_cq = CL_TRUE;
293  break;
294  }
295  }
296  g_assert(evt_in_cq);
297  }
298 
299  /* Clear events in queue and check that they were really destroyed. */
300  ccl_queue_gc(cq);
301  ccl_queue_iter_event_init(cq);
302  g_assert(ccl_queue_iter_event_next(cq) == NULL);
303 
304  /* Release wrappers. */
305  ccl_buffer_destroy(buf);
306  ccl_queue_destroy(cq);
307  ccl_context_destroy(ctx);
308 
309  /* Confirm that memory allocated by wrappers has been properly
310  * freed. */
311  g_assert(ccl_wrapper_memcheck());
312 
313 }
