35 /* Test utility macro. Presents either the required information, or
36  * the error message, if it occurred. Also frees the error object if
37  * an error occurred. */
38 #define ccl_test_platforms_msg(base_msg, format, ...) \
39  if (err == NULL) { \
40  g_snprintf(info_str, CCL_TEST_PLATFORMS_MAXINFOSTR, \
41  format, ##__VA_ARGS__); \
42  } else { \
43  g_snprintf(info_str, CCL_TEST_PLATFORMS_MAXINFOSTR, \
44  "%s", err->message); \
45  g_clear_error(&err); \
46  } \
47  g_debug("%s %s", base_msg, info_str);

49 static void ccl_test_platforms_check_error(GError** err) {
50  gboolean status;
51  if (*err == NULL) {
52  status = TRUE;
53  } else if (((*err)->domain == CCL_ERROR)
54  && ((*err)->code == CCL_ERROR_INFO_UNAVAILABLE_OCL)) {
55  status = TRUE;
56  } else if (((*err)->domain == CCL_OCL_ERROR)
57  && ((*err)->code == CL_INVALID_VALUE)) {
58  status = TRUE;
59  } else {
60  status = FALSE;
61  }
62  g_assert(status);
63 }
