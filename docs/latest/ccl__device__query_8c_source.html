50 static char* ccl_devquery_format_hex(CCLWrapperInfo* info,
51  char* out, size_t size, const char* units) {
52 
53  GString* str = g_string_new("0x");
54  gboolean start = FALSE;
55  gchar val;
56 
57  for (gint i = (gint) info->size - 1; i >= 0 ; i--) {
58  val = ((cl_char*) info->value)[i];
59  if (val) start = TRUE;
60  if (start)
61  g_string_append_printf(str, "%.2x", val);
62  }
63  if (units && units[0])
64  g_string_append_printf(str, " %s", units);
65 
66  g_snprintf(out, (gulong) size, "%s", str->str);
67  g_string_free(str, TRUE);
68  return out;
69 
70 }

91 #define ccl_devquery_format_bytes(spec) \
92  if (bytes < 1024) \
93  g_snprintf(out, (gulong) size, "%" spec " bytes", bytes); \
94  else if (bytes < 1048576) \
95  g_snprintf(out, (gulong) size, "%.1lf KiB (%" spec " bytes)", \
96  bytes / 1024.0, bytes); \
97  else if (bytes < 1073741824) \
98  g_snprintf(out, (gulong) size, "%.1lf MiB (%" spec " bytes)", \
99  bytes / (1024.0 * 1024), bytes); \
100  else \
101  g_snprintf(out, (gulong) size, "%.1lf GiB (%" spec " bytes)", \
102  bytes / (1024.0 * 1024 * 1024), bytes);

291 static char* ccl_devquery_format_partprop(CCLWrapperInfo* info,

293 
294  CCL_UNUSED(units);

300  switch (pp[i]) {
301  case CL_DEVICE_PARTITION_EQUALLY:

307  case CL_DEVICE_PARTITION_BY_AFFINITY_DOMAIN:

309  break;
310  case CL_DEVICE_PARTITION_EQUALLY_EXT:
311  g_string_append_printf(str, "EQUALLY_EXT ");
312  break;
313  case CL_DEVICE_PARTITION_BY_COUNTS_EXT:
314  g_string_append_printf(str, "BY_COUNTS_EXT ");
315  break;
316  case CL_DEVICE_PARTITION_BY_NAMES_EXT:
317  g_string_append_printf(str, "BY_NAMES_EXT ");

319  case CL_DEVICE_PARTITION_BY_AFFINITY_DOMAIN_EXT:

876 static int ccl_devquery_get_index(const char* name) {
877 
878  /* Make sure name is not NULL. */
879  g_return_val_if_fail(name != NULL, -1);
880 
881  /* Search indexes. */
882  gint idx_middle;
883  guint idx_start, idx_end;
884 
885  /* String comparison result. */
886  gint cmp_res;
887 
888  /* Found info flag. */
889  gboolean found = FALSE;
890 
891  /* Binary search. */
892  idx_start = 0;
893  idx_end = ccl_devquery_info_map_size - 1;
894  while (idx_end >= idx_start) {
895  idx_middle = (idx_start + idx_end) / 2;
896  cmp_res = g_strcmp0(
897  name, ccl_devquery_info_map[idx_middle].param_name);
898  if (cmp_res == 0) {
899  found = TRUE;
900  break;
901  }
902  if (cmp_res > 0)
903  idx_start = idx_middle + 1;
904  else
905  idx_end = idx_middle - 1;
906  }
907 
908  /* Return result */
909  return found ? idx_middle : -1;
910 }
