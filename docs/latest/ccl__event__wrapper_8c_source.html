653 CCLEventWaitList* ccl_event_wait_list_add(
654  CCLEventWaitList* evt_wait_lst, ...) {
655 
656  /* Check that evt_wait_lst is not NULL. */
657  g_return_val_if_fail(evt_wait_lst != NULL, NULL);
658 
659  /* Variable argument list. */
660  va_list al;
661 
662  /* Current event wrapper object. */
663  CCLEvent* evt;
664 
665  /* Initialize list if required. */
666  if (*evt_wait_lst == NULL)
667  *evt_wait_lst = g_ptr_array_new();
668 
669  /* Initialize variable argument list. */
670  va_start(al, evt_wait_lst);
671 
672  /* Get arguments (i.e. event wrapper objects). */
673  while ((evt = va_arg(al, CCLEvent*)) != NULL) {
674 
675  /* Add event wrapper to array. */
676  g_ptr_array_add(*evt_wait_lst, ccl_event_unwrap(evt));
677 
678  }
679 
680  /* Finalize variable argument list. */
681  va_end(al);
682 
683  /* Signal bug if no events have been given. */
684  g_return_val_if_fail((*evt_wait_lst)->len > 0, NULL);
685 
686  /* Return event wait list. */
687  return evt_wait_lst;
688 
689 }
