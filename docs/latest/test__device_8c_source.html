33 static void sub_devices_test() {
34 
35 #ifdef CL_VERSION_1_2
36 
37  /* Test variables. */
38  CCLContext* ctx = NULL;
39  CCLDevice* pdev = NULL;
40  GError* err = NULL;
41  cl_uint ocl_ver;
42  cl_device_partition_property* dpp;
43  cl_uint i;
44  cl_bool supported;
45  CCLDevice* const* subdevs;
46  cl_uint num_subdevs;
47  cl_uint max_subdevs;
48  cl_uint cu;
49  cl_uint subcu;
50  cl_device_id parent_device;
51 
52  /* Get the test context with the pre-defined device. */
53  ctx = ccl_test_context_new(&err);
54  g_assert_no_error(err);
55 
56  /* Check OpenCL version of the platform. */
57  ocl_ver = ccl_context_get_opencl_version(ctx, &err);
58  g_assert_no_error(err);
59 
60  /* This test can only be performed with devices supporting OpenCL
61  * 1.2 or higher. */
62  if (ocl_ver < 120) {
63  g_test_message("OpenCL version of parent device does not "\
64  "support sub-devices. Sub-devices test not performed.");
65  return;
66  }
67 
68  /* Get parent device. */
69  pdev = ccl_context_get_device(ctx, 0, &err);
70  g_assert_no_error(err);
71 
72  /* Get device partition properties. */
73  dpp = ccl_device_get_info_array(pdev, CL_DEVICE_PARTITION_PROPERTIES,
74  cl_device_partition_property*, &err);
75  g_assert_no_error(err);
76 
77  /* Get maximum number sub-devices. */
78  max_subdevs = ccl_device_get_info_scalar(
79  pdev, CL_DEVICE_PARTITION_MAX_SUB_DEVICES, cl_uint, &err);
80  g_assert_no_error(err);
81 
82  /* Test partition equally, if supported by device. */
83  supported = CL_FALSE;
84  for (i = 0; dpp[i] != 0; ++i) {
85  if (dpp[i] == CL_DEVICE_PARTITION_EQUALLY) {
86  supported = CL_TRUE;
87  break;
88  }
89  }
90  if (supported) {
91 
92  /* Find an appropriate number of compute units for each
93  * sub-device. */
94  for (i = 8; (cu = max_subdevs / i) == 0; i /= 2);
95 
96  /* Set partition properties. */
97  const cl_device_partition_property eqprop[] =
98  {CL_DEVICE_PARTITION_EQUALLY, cu, 0};
99 
100  /* Partition device. */
101  subdevs = ccl_device_create_subdevices(
102  pdev, eqprop, &num_subdevs, &err);
103  g_assert_no_error(err);
104 
105  /* Check sub-devices. */
106  for (i = 0; i < num_subdevs; ++i) {
107 
108  /* Check the number of compute units. */
109  subcu = ccl_device_get_info_scalar(subdevs[i],
110  CL_DEVICE_MAX_COMPUTE_UNITS, cl_uint, &err);
111  g_assert_no_error(err);
112  g_assert_cmpuint(subcu, ==, cu);
113 
114  /* Check the parent device. */
115  parent_device = ccl_device_get_info_scalar(
116  subdevs[i], CL_DEVICE_PARENT_DEVICE, cl_device_id, &err);
117  g_assert_no_error(err);
118  g_assert_cmphex(GPOINTER_TO_UINT(parent_device), ==,
119  GPOINTER_TO_UINT(ccl_device_unwrap(pdev)));
120  }
121 
122  /* Check that the last position is NULL. */
123  g_assert_cmphex(
124  GPOINTER_TO_UINT(subdevs[i]), ==, GPOINTER_TO_UINT(NULL));
125 
126  }
127 
128  /* Test partition by counts, if supported by device. */
129  supported = CL_FALSE;
130  for (i = 0; dpp[i] != 0; ++i) {
131  if (dpp[i] == CL_DEVICE_PARTITION_BY_COUNTS) {
132  supported = CL_TRUE;
133  break;
134  }
135  }
136  if (supported) {
137 
138  /* Allocate partition properties array and initialize it. */
139  cl_device_partition_property* ctprop = g_slice_alloc0(
140  (max_subdevs + 3) * sizeof(cl_device_partition_property));
141  ctprop[0] = CL_DEVICE_PARTITION_BY_COUNTS;
142 
143  /* Find an appropriate number of compute units for each
144  * sub-device. */
145  cu = max_subdevs / 2;
146  cl_uint total_cu = 0, total_cu_check = 0;
147  if (cu == 0) {
148  total_cu = 1;
149  ctprop[1] = max_subdevs;
150  ctprop[2] = CL_DEVICE_PARTITION_BY_COUNTS_LIST_END;
151  } else {
152  for (i = 1; (i <= max_subdevs) && (cu > 0); ++i) {
153  total_cu += cu;
154  ctprop[i] = cu;
155  cu /= 2;
156  }
157  ctprop[i] = CL_DEVICE_PARTITION_BY_COUNTS_LIST_END;
158  }
159 
160  /* Partition device. */
161  subdevs = ccl_device_create_subdevices(
162  pdev, ctprop, &num_subdevs, &err);
163  g_assert_no_error(err);
164 
165  /* Check sub-devices. */
166  for (i = 0; i < num_subdevs; ++i) {
167 
168  /* Check the number of compute units. */
169  subcu = ccl_device_get_info_scalar(subdevs[i],
170  CL_DEVICE_MAX_COMPUTE_UNITS, cl_uint, &err);
171  g_assert_no_error(err);
172  total_cu_check += subcu;
173 
174  /* Check the parent device. */
175  parent_device = ccl_device_get_info_scalar(
176  subdevs[i], CL_DEVICE_PARENT_DEVICE, cl_device_id, &err);
177  g_assert_no_error(err);
178  g_assert_cmphex(GPOINTER_TO_UINT(parent_device), ==,
179  GPOINTER_TO_UINT(ccl_device_unwrap(pdev)));
180  }
181 
182  /* Check that the total number of compute units is as
183  * expected. */
184  g_assert_cmpuint(total_cu_check, ==, total_cu);
185 
186  /* Check that the last position is NULL. */
187  g_assert_cmphex(
188  GPOINTER_TO_UINT(subdevs[i]), ==, GPOINTER_TO_UINT(NULL));
189 
190  /* Release memory associated with partition properties array. */
191  g_slice_free1(
192  (max_subdevs + 3) * sizeof(cl_device_partition_property),
193  ctprop);
194  }
195 
196  /* Destroy stuff. */
197  ccl_context_destroy(ctx);
198 
199  /* Confirm that memory allocated by wrappers has been properly
200  * freed. */
201  g_assert(ccl_wrapper_memcheck());
202 
203 #else
204  g_test_message("OpenCL version of platform does not support " \
205  "sub-devices. Sub-devices test not performed.");
206 #endif
207 
208 }
