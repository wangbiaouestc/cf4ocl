80 CCLWrapper* ccl_wrapper_new(void* cl_object, size_t size) {
81 
82  /* Make sure OpenCL object is not NULL. */
83  g_return_val_if_fail(cl_object != NULL, NULL);
84 
85  /* The new wrapper object. */
86  CCLWrapper* w;
87 
88  /* Lock access to table of all existing wrappers. */
89  G_LOCK(wrappers);
90 
91  /* If table of all existing wrappers is not yet initialized,
92  * initialize it. */
93  if (wrappers == NULL) {
94  wrappers = g_hash_table_new_full(
95  g_direct_hash, g_direct_equal, NULL, NULL);
96  }
97 
98  /* Check if requested wrapper already exists, and get it if so. */
99  w = g_hash_table_lookup(wrappers, cl_object);
100  if (w == NULL) {
101  /* Wrapper doesn't yet exist, create it. */
102  w = (CCLWrapper*) g_slice_alloc0(size);
103  w->cl_object = cl_object;
104  /* Initialize info table. */
105  w->info = g_slice_new0(struct ccl_wrapper_info_table);
106  g_mutex_init(&w->info->mutex);
107  /* Insert newly created wrapper in table of all existing
108  * wrappers. */
109  g_hash_table_insert(wrappers, cl_object, w);
110  }
111 
112  /* Increase reference count of wrapper. */
113  ccl_wrapper_ref(w);
114 
115  /* Unlock access to table of all existing wrappers. */
116  G_UNLOCK(wrappers);
117 
118  /* Return requested wrapper. */
119  return w;
120 }

141 cl_bool ccl_wrapper_unref(CCLWrapper* wrapper, size_t size,
142  ccl_wrapper_release_fields rel_fields_fun,
143  ccl_wrapper_release_cl_object rel_cl_fun, GError** err) {
144 
145  /* Make sure wrapper object is not NULL. */
146  g_return_val_if_fail(wrapper != NULL, CL_FALSE);
147 
148  /* Make sure err is NULL or it is not set. */
149  g_return_val_if_fail(err == NULL || *err == NULL, CL_FALSE);
150 
151  /* Flag which indicates if wrapper was destroyed or not. */
152  cl_bool destroyed = CL_FALSE;
153 
154  /* OpenCL status flag. */
155  cl_int ocl_status;
156 
157  /* Decrement reference count and check if it reaches 0. */
158  if (g_atomic_int_dec_and_test(&wrapper->ref_count)) {
159 
160  /* Ref. count reached 0, so wrapper will be destroyed. */
161  destroyed = CL_TRUE;
162 
163  /* Release the OpenCL wrapped object. */
164  if (rel_cl_fun != NULL) {
165  ocl_status = rel_cl_fun(wrapper->cl_object);
166  if (ocl_status != CL_SUCCESS) {
167  g_set_error(err, CCL_OCL_ERROR, ocl_status,
168  "%s: unable to create release OpenCL object (OpenCL error %d: %s).",
169  CCL_STRD, ocl_status, ccl_err(ocl_status));
170  }
171  }
172 
173  /* Destroy table containing wrapped object information. */
174  if (wrapper->info->table != NULL) {
175  g_hash_table_destroy(wrapper->info->table);
176  }
177  if (wrapper->info->old_info != NULL) {
178  g_slist_free_full(wrapper->info->old_info,
179  (GDestroyNotify) ccl_wrapper_info_destroy);
180  }
181  g_mutex_clear(&wrapper->info->mutex);
182  g_slice_free(struct ccl_wrapper_info_table, wrapper->info);
183 
184  /* Remove wrapper from static table, release static table if
185  * empty. */
186  G_LOCK(wrappers);
187  g_hash_table_remove(wrappers, wrapper->cl_object);
188  if (g_hash_table_size(wrappers) == 0) {
189  g_hash_table_destroy(wrappers);
190  wrappers = NULL;
191  }
192  G_UNLOCK(wrappers);
193 
194  /* Destroy remaining wrapper fields. */
195  if (rel_fields_fun != NULL)
196  rel_fields_fun(wrapper);
197 
198  /* Destroy wrapper. */
199  g_slice_free1(size, wrapper);
200 
201  }
202 
203  /* Return flag indicating if wrapper was destroyed. */
204  return destroyed;
205 
206 }

220 void ccl_wrapper_add_info(CCLWrapper* wrapper, cl_uint param_name,
221  CCLWrapperInfo* info) {
222 
223  /* Make sure wrapper is not NULL. */
224  g_return_if_fail(wrapper != NULL);
225  /* Make sure info is not NULL. */
226  g_return_if_fail(info != NULL);
227 
228  /* Lock access to info table. */
229  g_mutex_lock(&wrapper->info->mutex);
230 
231  /* If information table is not yet initialized, then
232  * initialize it. */
233  if (wrapper->info->table == NULL) {
234  wrapper->info->table = g_hash_table_new_full(
235  g_direct_hash, g_direct_equal,
236  NULL, (GDestroyNotify) ccl_wrapper_info_destroy);
237  }
238 
239  /* Check if information with same key as already present in
240  * table... */
241  if (g_hash_table_contains(
242  wrapper->info->table, GUINT_TO_POINTER(param_name))) {
243 
244  /* ...if so, move this information to the old information
245  * table. */
246 
247  /* Get existing information... */
248  CCLWrapperInfo* info_old =
249  (CCLWrapperInfo*) g_hash_table_lookup(
250  wrapper->info->table, GUINT_TO_POINTER(param_name));
251 
252  /* ...and put it in table of old information. */
253  wrapper->info->old_info =
254  g_slist_prepend(wrapper->info->old_info, info_old);
255 
256  /* Remove old info from info table without destroying it. */
257  g_hash_table_steal(
258  wrapper->info->table, GUINT_TO_POINTER(param_name));
259 
260  }
261 
262  /* Keep new information in information table. */
263  g_hash_table_insert(wrapper->info->table,
264  GUINT_TO_POINTER(param_name), info);
265 
266  /* Unlock access to info table. */
267  g_mutex_unlock(&wrapper->info->mutex);
268 
269 }

398 CCLWrapperInfo* ccl_wrapper_get_info(CCLWrapper* wrapper1,
399  CCLWrapper* wrapper2, cl_uint param_name, size_t min_size,
400  ccl_wrapper_info_fp info_fun, cl_bool use_cache, GError** err) {
401 
402  /* Make sure err is NULL or it is not set. */
403  g_return_val_if_fail((err) == NULL || *(err) == NULL, NULL);
404 
405  /* Make sure wrapper1 is not NULL. */
406  g_return_val_if_fail(wrapper1 != NULL, NULL);
407 
408  /* Information object. */
409  CCLWrapperInfo* info = NULL;
410 
411  /* Does info table cache contain requested information? */
412  gboolean contains;
413 
414  /* Check if info table cache contains requested information. */
415  g_mutex_lock(&wrapper1->info->mutex);
416  contains = wrapper1->info->table != NULL
417  ? g_hash_table_contains(
418  wrapper1->info->table, GUINT_TO_POINTER(param_name))
419  : FALSE;
420  g_mutex_unlock(&wrapper1->info->mutex);
421 
422  /* Check if it is required to query OpenCL object, i.e. if info
423  * table cache is not to be used or info table cache does not
424  * contain requested info. */
425  if ((!use_cache) || (!contains)) {
426 
427  /* Let's query OpenCL object.*/
428  cl_int ocl_status;
429  /* Size of device information in bytes. */
430  size_t size_ret = 0;
431 
432  /* Get size of information. */
433  ocl_status = (wrapper2 == NULL)
434  ? ((ccl_wrapper_info_fp1) info_fun)(wrapper1->cl_object,
435  param_name, 0, NULL, &size_ret)
436  : ((ccl_wrapper_info_fp2) info_fun)(wrapper1->cl_object,
437  wrapper2->cl_object, param_name, 0, NULL, &size_ret);
438 
439  /* Avoid bug in Apple OpenCL implementation. */
440 #if defined(__APPLE__) || defined(__MACOSX)
441  if ((ocl_status == CL_INVALID_VALUE)
442  && (info_fun == (ccl_wrapper_info_fp) clGetEventProfilingInfo))
443  ocl_status = CL_SUCCESS;
444 #endif
445 
446  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
447  CL_SUCCESS != ocl_status, ocl_status, error_handler,
448  "%s: get info [size] (OpenCL error %d: %s).",
449  CCL_STRD, ocl_status, ccl_err(ocl_status));
450  ccl_if_err_create_goto(*err, CCL_ERROR, size_ret == 0,
451  CCL_ERROR_INFO_UNAVAILABLE_OCL, error_handler,
452  "%s: the requested info is unavailable (info size is 0).",
453  CCL_STRD);
454 
455  /* Allocate memory for information. */
456  info = ccl_wrapper_info_new(size_ret);
457 
458  /* Get information. */
459  ocl_status = (wrapper2 == NULL)
460  ? ((ccl_wrapper_info_fp1) info_fun)(wrapper1->cl_object,
461  param_name, size_ret, info->value, NULL)
462  : ((ccl_wrapper_info_fp2) info_fun)(wrapper1->cl_object,
463  wrapper2->cl_object, param_name, size_ret, info->value,
464  NULL);
465  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
466  CL_SUCCESS != ocl_status, ocl_status, error_handler,
467  "%s: get context info [info] (OpenCL error %d: %s).",
468  CCL_STRD, ocl_status, ccl_err(ocl_status));
469 
470  /* Keep information in information table. */
471  ccl_wrapper_add_info(wrapper1, param_name, info);
472 
473  } else {
474 
475  /* Requested info is already present in the info table,
476  * retrieve it from there. */
477  g_mutex_lock(&wrapper1->info->mutex);
478  info = g_hash_table_lookup(
479  wrapper1->info->table, GUINT_TO_POINTER(param_name));
480  g_mutex_unlock(&wrapper1->info->mutex);
481 
482  }
483 
484  /* If we got here, everything is OK. */
485  g_assert(err == NULL || *err == NULL);
486  goto finish;
487 
488 error_handler:
489  /* If we got here there was an error, verify that it is so. */
490  g_assert(err == NULL || *err != NULL);
491 
492  /* In case of error, return an all-zeros info if min_size is > 0. */
493  if (min_size > 0) {
494  if (info != NULL) ccl_wrapper_info_destroy(info);
495  info = ccl_wrapper_info_new(min_size);
496  ccl_wrapper_add_info(wrapper1, param_name, info);
497  }
498 
499 finish:
500 
501  /* Return the requested information. */
502  return info;
503 }
