178 CCLDevice* const* ccl_device_create_subdevices(CCLDevice* dev,
179  const cl_device_partition_property *properties,
180  cl_uint *num_devs_ret, GError** err) {
181 
182  /* Make sure device wrapper object is not NULL. */
183  g_return_val_if_fail(dev != NULL, NULL);
184 
185  /* OpenCL function return status. */
186  cl_int ocl_status;
187  /* Number of sub-devices the partition may yield. */
188  cl_uint num_devs;
189  /* Array for holding sub-devices. */
190  void** subdevs = NULL;
191 
192  /* How many sub-devices will this partition yield? */
193  ocl_status = clCreateSubDevices(
194  ccl_device_unwrap(dev), properties, 0, NULL, &num_devs);
195  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
196  CL_SUCCESS != ocl_status, ocl_status, error_handler,
197  "%s: unable to determine how many sub-devices can be created " \
198  "(OpenCL error %d: %s).",
199  CCL_STRD, ocl_status, ccl_err(ocl_status));
200 
201  /* Create array for holding sub-devices. The last element should
202  * be NULL. */
203  subdevs = g_new0(void*, num_devs + 1);
204 
205  /* Create sub-devices. */
206  ocl_status = clCreateSubDevices(
207  ccl_device_unwrap(dev), properties, num_devs,
208  (cl_device_id*) subdevs, NULL);
209  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
210  CL_SUCCESS != ocl_status, ocl_status, error_handler,
211  "%s: unable to create sub-devices (OpenCL error %d: %s).",
212  CCL_STRD, ocl_status, ccl_err(ocl_status));
213 
214  /* Wrap sub-devices. */
215  for (cl_uint i = 0; subdevs[i] != NULL; ++i)
216  subdevs[i] = ccl_device_new_wrap(subdevs[i]);
217 
218  /* If we got here, everything is OK. */
219  g_assert(err == NULL || *err == NULL);
220  if (num_devs_ret != NULL) *num_devs_ret = num_devs;
221  dev->subdev_arrays = g_slist_prepend(dev->subdev_arrays, subdevs);
222  goto finish;
223 
224 error_handler:
225  /* If we got here there was an error, verify that it is so. */
226  g_assert(err == NULL || *err != NULL);
227 
228  /* An error occurred, return NULL to signal it. */
229  if (num_devs_ret != NULL) *num_devs_ret = 0;
230  if (subdevs != NULL) g_free(subdevs);
231  subdevs = NULL;
232 
233 finish:
234 
235  /* Return event. */
236  return (CCLDevice* const*) subdevs;
237 }
