281 static void ref_unref_test() {
282 
283  CCLContext* ctx = NULL;
284  GError* err = NULL;
285  CCLProgram* prg = NULL;
286  CCLKernel* krnl1 = NULL;
287  CCLKernel* krnl2 = NULL;
288 
289  /* Get some context. */
290  ctx = ccl_test_context_new(&err);
291  g_assert_no_error(err);
292 
293  /* Create a program from source. */
294  prg = ccl_program_new_from_source(
295  ctx, CCL_TEST_KERNEL_CONTENT, &err);
296  g_assert_no_error(err);
297 
298  /* Build program. */
299  ccl_program_build(prg, NULL, &err);
300  g_assert_no_error(err);
301 
302  /* Get kernel wrapper from program (will be the instance kept in the
303  * program wrapper). */
304  krnl1 = ccl_program_get_kernel(prg, CCL_TEST_KERNEL_NAME, &err);
305  g_assert_no_error(err);
306 
307  /* Create another kernel wrapper for the same kernel. This should
308  * yield a different object because we're not getting it from
309  * the program wrapper. */
310  krnl2 = ccl_kernel_new(prg, CCL_TEST_KERNEL_NAME, &err);
311  g_assert_no_error(err);
312 
313  /* Check that they're different. */
314  g_assert_cmphex(
315  GPOINTER_TO_UINT(krnl1), !=, GPOINTER_TO_UINT(krnl2));
316 
317  /* Check that each has a ref count of 1. */
318  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) krnl1), ==, 1);
319  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) krnl2), ==, 1);
320 
321  /* Increment the ref count of the directly created kernel. */
322  ccl_kernel_ref(krnl2);
323  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) krnl1), ==, 1);
324  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) krnl2), ==, 2);
325 
326  /* Get rid of the directly created kernel. */
327  ccl_kernel_unref(krnl2);
328  ccl_kernel_unref(krnl2);
329 
330  /* Get kernel wrapper from program again (will be the instance kept
331  * in the program wrapper). */
332  krnl2 = ccl_program_get_kernel(prg, CCL_TEST_KERNEL_NAME, &err);
333  g_assert_no_error(err);
334 
335  /* Check that it's the same as krnl1. */
336  g_assert_cmphex(
337  GPOINTER_TO_UINT(krnl1), ==, GPOINTER_TO_UINT(krnl2));
338 
339  /* Destroy remaining stuff. */
340  ccl_program_destroy(prg);
341  ccl_context_destroy(ctx);
342 
343  /* Confirm that memory allocated by wrappers has been properly
344  * freed. */
345  g_assert(ccl_wrapper_memcheck());
346 
347 }

360 static void check_dev_limits(CCLDevice* dev, cl_uint dims, size_t* lws) {
361 
362  /* Error handling object. */
363  GError* err = NULL;
364 
365  /* Max device workgroup size. */
366  size_t max_wgsize = ccl_device_get_info_scalar(
367  dev, CL_DEVICE_MAX_WORK_GROUP_SIZE, size_t, &err);
368  g_assert_no_error(err);
369 
370  /* Max device workitem sizes. */
371  size_t* max_wisizes = ccl_device_get_info_array(
372  dev, CL_DEVICE_MAX_WORK_ITEM_SIZES, size_t*, &err);
373  g_assert_no_error(err);
374 
375  /* Suggested work group size. */
376  size_t wgsize = 1;
377 
378  /* Check dimension by dimension. */
379  for (cl_uint i = 0; i < dims; ++i) {
380 
381  /* Check workitem size. */
382  g_assert_cmpuint(lws[i], <=, max_wisizes[i]);
383 
384  /* Update work group size with size in current dimension. */
385  wgsize *= lws[i];
386  }
387 
388  /* Check work group size. */
389  g_assert_cmpuint(wgsize, <=, max_wgsize);
390 
391 }
