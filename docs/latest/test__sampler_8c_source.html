34 static void create_info_destroy_test() {
35 
36  /* Test variables. */
37  CCLContext* ctx = NULL;
38  CCLSampler* s = NULL;
39  cl_sampler sampler = NULL;
40  GError* err = NULL;
41  cl_int ocl_status;
42  const cl_sampler_properties sampler_properties[] = {
43  CL_SAMPLER_NORMALIZED_COORDS, CL_TRUE,
44  CL_SAMPLER_ADDRESSING_MODE, CL_ADDRESS_NONE,
45  CL_SAMPLER_FILTER_MODE, CL_FILTER_NEAREST,
46  0};
47 
48  /* Get the test context with the pre-defined device. */
49  ctx = ccl_test_context_new(&err);
50  g_assert_no_error(err);
51 
52  /* Test three ways to create a sampler. */
53  for (cl_uint i = 0; i < 3; ++i) {
54 
55  /* Create sampler wrapper. */
56  switch (i) {
57  case 0:
58  /* Create sampler using "simple" constructor. */
59  s = ccl_sampler_new(ctx, CL_TRUE, CL_ADDRESS_NONE,
60  CL_FILTER_NEAREST, &err);
61  g_assert_no_error(err);
62  break;
63  case 1:
64  /* Using the "full" constructor. */
65  s = ccl_sampler_new_full(ctx, sampler_properties, &err);
66  g_assert_no_error(err);
67  break;
68  case 2:
69  /* Using the "wrap" constructor. */
70  CCL_BEGIN_IGNORE_DEPRECATIONS
71  sampler = clCreateSampler(ccl_context_unwrap(ctx),
72  CL_TRUE, CL_ADDRESS_NONE, CL_FILTER_NEAREST,
73  &ocl_status);
74  g_assert_cmpint(ocl_status, ==, CL_SUCCESS);
75  CCL_END_IGNORE_DEPRECATIONS
76  s = ccl_sampler_new_wrap(sampler);
77  g_assert_cmphex(GPOINTER_TO_UINT(sampler), ==,
78  GPOINTER_TO_UINT(ccl_sampler_unwrap(s)));
79  break;
80  }
81 
82  /* Get some info and check if the return value is as expected. */
83  cl_addressing_mode am;
84  am = ccl_sampler_get_info_scalar(
85  s, CL_SAMPLER_ADDRESSING_MODE, cl_addressing_mode, &err);
86  g_assert_no_error(err);
87  g_assert_cmpuint(am, ==, CL_ADDRESS_NONE);
88 
89  cl_filter_mode fm;
90  fm = ccl_sampler_get_info_scalar(
91  s, CL_SAMPLER_FILTER_MODE, cl_filter_mode, &err);
92  g_assert_no_error(err);
93  g_assert_cmpuint(fm, ==, CL_FILTER_NEAREST);
94 
95  cl_bool nc;
96  nc = ccl_sampler_get_info_scalar(
97  s, CL_SAMPLER_NORMALIZED_COORDS, cl_bool, &err);
98  g_assert_no_error(err);
99  g_assert_cmpuint(nc, ==, CL_TRUE);
100 
101  cl_context context;
102  context = ccl_sampler_get_info_scalar(
103  s, CL_SAMPLER_CONTEXT, cl_context, &err);
104  g_assert_no_error(err);
105  g_assert_cmphex(GPOINTER_TO_UINT(context), ==,
106  GPOINTER_TO_UINT(ccl_context_unwrap(ctx)));
107 
108  /* Destroy sampler. */
109  ccl_sampler_destroy(s);
110  }
111 
112  /* Destroy context. */
113  ccl_context_destroy(ctx);
114 
115  /* Confirm that memory allocated by wrappers has been properly
116  * freed. */
117  g_assert(ccl_wrapper_memcheck());
118 
119 }

185 static void ref_unref_test() {
186 
187  /* Test variables. */
188  CCLContext* ctx = NULL;
189  CCLSampler* s = NULL;
190  GError* err = NULL;
191 
192  /* Get the test context with the pre-defined device. */
193  ctx = ccl_test_context_new(&err);
194  g_assert_no_error(err);
195 
196  /* Create sampler. */
197  s = ccl_sampler_new(ctx, CL_TRUE, CL_ADDRESS_CLAMP,
198  CL_FILTER_NEAREST, &err);
199  g_assert_no_error(err);
200 
201  /* Increase sampler reference count. */
202  ccl_sampler_ref(s);
203 
204  /* Check that sampler ref count is 2. */
205  g_assert_cmpuint(2, ==, ccl_wrapper_ref_count((CCLWrapper*) s));
206 
207  /* Unref sampler. */
208  ccl_sampler_unref(s);
209 
210  /* Check that sampler ref count is 1. */
211  g_assert_cmpuint(1, ==, ccl_wrapper_ref_count((CCLWrapper*) s));
212 
213  /* Destroy stuff. */
214  ccl_sampler_unref(s);
215  ccl_context_destroy(ctx);
216 
217  /* Confirm that memory allocated by wrappers has been properly
218  * freed. */
219  g_assert(ccl_wrapper_memcheck());
220 }
