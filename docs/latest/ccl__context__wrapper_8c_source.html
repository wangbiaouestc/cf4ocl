109 static cl_context_properties* ccl_context_properties_default(
110  const cl_context_properties* properties,
111  cl_device_id device, GError** err) {
112 
113  /* Make sure device is not NULL. */
114  g_return_val_if_fail(device != NULL, NULL);
115  /* Make sure err is NULL or it is not set. */
116  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
117 
118  /* Context properties. */
119  cl_context_properties* ctx_props = NULL;
120 
121  /* The OpenCL platform object. */
122  cl_platform_id platform;
123 
124  /* OpenCL function call status. */
125  cl_int ocl_status;
126 
127  /* Check if the original const properties object is NULL... */
128  if (properties == NULL) {
129 
130  /* ...if so, create a default set of context properties. */
131 
132  /* Allocate memory for default properties. */
133  ctx_props = g_slice_alloc(3 * sizeof(cl_context_properties));
134 
135  /* Get context platform using first device. */
136  ocl_status = clGetDeviceInfo(device, CL_DEVICE_PLATFORM,
137  sizeof(cl_platform_id), &platform, NULL);
138  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
139  CL_SUCCESS != ocl_status, ocl_status, error_handler,
140  "%s: unable to get platform from device (OpenCL error %d: %s).",
141  CCL_STRD, ocl_status, ccl_err(ocl_status));
142 
143  /* Set context properties using discovered platform. */
144  ctx_props[0] = CL_CONTEXT_PLATFORM;
145  ctx_props[1] = (cl_context_properties) platform;
146  ctx_props[2] = 0;
147 
148  } else {
149 
150  /* If properties parameter is not NULL, use it instead. */
151  ctx_props = (cl_context_properties*) properties;
152  }
153 
154  /* If we got here, everything is OK. */
155  g_assert(err == NULL || *err == NULL);
156  goto finish;
157 
158 error_handler:
159  /* If we got here there was an error, verify that it is so. */
160  g_assert(err == NULL || *err != NULL);
161 
162 finish:
163 
164  /* Return properties. */
165  return ctx_props;
166

320 CCL_EXPORT
321 CCLContext* ccl_context_new_from_devices_full(
322  const cl_context_properties* properties, cl_uint num_devices,
323  CCLDevice* const* devices, ccl_context_callback pfn_notify,
324  void* user_data, GError** err) {
325 
326  /* Make sure number of devices is not zero. */
327  g_return_val_if_fail(num_devices > 0, NULL);
328  /* Make sure device array is not NULL. */
329  g_return_val_if_fail(devices != NULL, NULL);
330  /* Make sure err is NULL or it is not set. */
331  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
332 
333  /* Array of unwrapped devices. */
334  cl_device_id* cl_devices = NULL;
335  /* Context properties, in case the properties parameter is NULL. */
336  cl_context_properties* ctx_props = NULL;
337  /* Return status of OpenCL function calls. */
338  cl_int ocl_status;
339  /* OpenCL context. */
340  cl_context context = NULL;
341  /* New context wrapper. */
342  CCLContext* ctx = NULL;
343  /* Error reporting object. */
344  GError* err_internal = NULL;
345 
346  /* Allocate memory for devices. */
347  cl_devices = (cl_device_id*) g_slice_alloc(sizeof(cl_device_id) * num_devices);
348 
349  /* Unwrap devices. */
350  for (guint i = 0; i < num_devices; i++)
351  cl_devices[i] = ccl_device_unwrap(devices[i]);
352 
353  /* Get a set of default context properties, if required. */
354  ctx_props = ccl_context_properties_default(
355  properties, cl_devices[0], &err_internal);
356 
357  /* Create OpenCL context. */
358  context = clCreateContext(
359  (const cl_context_properties*) ctx_props, num_devices,
360  (const cl_device_id*) cl_devices, pfn_notify, user_data,
361  &ocl_status);
362  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
363  CL_SUCCESS != ocl_status, ocl_status, error_handler,
364  "%s: unable to create cl_context (OpenCL error %d: %s).",
365  CCL_STRD, ocl_status, ccl_err(ocl_status));
366 
367  /* Wrap OpenCL context. */
368  ctx = ccl_context_new_wrap(context);
369 
370  /* If we got here, everything is OK. */
371  g_assert(err == NULL || *err == NULL);
372  goto finish;
373 
374 error_handler:
375  /* If we got here there was an error, verify that it is so. */
376  g_assert(err == NULL || *err != NULL);
377 
378  /* Destroy what was built for the context wrapper. */
379  ccl_context_destroy(ctx);
380  ctx = NULL;
381 
382 finish:
383 
384  /* Free stuff. */
385  ccl_context_properties_default_free(properties, ctx_props);
386  g_slice_free1(sizeof(cl_device_id) * num_devices, cl_devices);
387 
388  /* Return result of function call. */
389  return ctx;
390 
391 }

624 const cl_image_format* ccl_context_get_supported_image_formats(
625  CCLContext* ctx, cl_mem_flags flags, cl_mem_object_type image_type,
626  cl_uint* num_image_formats, GError** err) {
627 
628  /* Make sure ctx is not NULL. */
629  g_return_val_if_fail(ctx != NULL, NULL);
630  /* Make sure num_image_formats is not NULL. */
631  g_return_val_if_fail(num_image_formats != NULL, NULL);
632  /* Make sure err is NULL or it is not set. */
633  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
634 
635  /* Information object. We use it to keep image format information
636  * in the information table, so that it can be automatically
637  * destroyed when the context is destroyed. */
638  CCLWrapperInfo* info = NULL;
639 
640  /* Variable to return. */
641  const cl_image_format* image_formats = NULL;
642 
643  /* Let's query OpenCL object.*/
644  cl_int ocl_status;
645 
646  /* Get number of image formats. */
647  ocl_status = clGetSupportedImageFormats(ccl_context_unwrap(ctx),
648  flags, image_type, 0, NULL, num_image_formats);
649  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
650  CL_SUCCESS != ocl_status, ocl_status, error_handler,
651  "%s: get number of supported image formats (OpenCL error %d: %s).",
652  CCL_STRD, ocl_status, ccl_err(ocl_status));
653  ccl_if_err_create_goto(*err, CCL_ERROR,
654  *num_image_formats == 0, CCL_ERROR_OTHER, error_handler,
655  "%s: number of returned supported image formats is 0.",
656  CCL_STRD);
657 
658  /* Allocate memory for number of image formats. */
659  info = ccl_wrapper_info_new(
660  (*num_image_formats) * sizeof(cl_image_format));
661 
662  /* Get image formats. */
663  ocl_status = clGetSupportedImageFormats(ccl_context_unwrap(ctx),
664  flags, image_type, *num_image_formats,
665  (cl_image_format*) info->value, NULL);
666  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
667  CL_SUCCESS != ocl_status, ocl_status, error_handler,
668  "%s: get supported image formats (OpenCL error %d: %s).",
669  CCL_STRD, ocl_status, ccl_err(ocl_status));
670 
671  /* Keep information in information table for latter disposal. */
672  ccl_wrapper_add_info((CCLWrapper*) ctx, CL_IMAGE_FORMAT, info);
673 
674  /* If we got here, everything is OK. */
675  g_assert(err == NULL || *err == NULL);
676  image_formats = (const cl_image_format*) info->value;
677  goto finish;
678 
679 error_handler:
680  /* If we got here there was an error, verify that it is so. */
681  g_assert(err == NULL || *err != NULL);
682  *num_image_formats = 0;
683 
684 finish:
685 
686  /* Return requested supported image formats. */
687  return image_formats;
688 }
