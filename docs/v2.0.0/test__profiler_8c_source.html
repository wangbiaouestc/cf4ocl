35 static void create_add_destroy_test() {
36 
37  /* Test variables. */
38  GError* err = NULL;
39  CCLBuffer* buf1 = NULL;
40  CCLBuffer* buf2 = NULL;
41  CCLProf* prof = NULL;
42  CCLContext* ctx = NULL;
43  CCLDevice* d = NULL;
44  CCLQueue* cq1 = NULL;
45  CCLQueue* cq2 = NULL;
46  CCLEvent* evt = NULL;
47  CCLEventWaitList ewl = NULL;
48  size_t buf_size = 8 * sizeof(cl_short);
49  cl_short hbuf[8] = {1, 2, 3, 4, 5, 6, 7, 8};
50  cl_ulong duration, eff_duration;
51  double time_ellapsed;
52 
53  /* Create a new profile object. */
54  prof = ccl_prof_new();
55 
56  /* Get a context and a device. */
57  ctx = ccl_test_context_new(&err);
58  g_assert_no_error(err);
59 
60  d = ccl_context_get_device(ctx, 0, &err);
61  g_assert_no_error(err);
62 
63  /* Create two command queue wrappers. */
64  cq1 = ccl_queue_new(ctx, d, CL_QUEUE_PROFILING_ENABLE, &err);
65  g_assert_no_error(err);
66 
67  cq2 = ccl_queue_new(ctx, d, CL_QUEUE_PROFILING_ENABLE, &err);
68  g_assert_no_error(err);
69 
70  /* Create device buffers. */
71  buf1 = ccl_buffer_new(ctx, CL_MEM_READ_ONLY, buf_size, NULL, &err);
72  g_assert_no_error(err);
73  buf2 = ccl_buffer_new(ctx, CL_MEM_READ_WRITE, buf_size, NULL, &err);
74  g_assert_no_error(err);
75 
76  /* Start profile object timer. */
77  ccl_prof_start(prof);
78 
79  /* Transfer data to buffer. */
80  evt = ccl_buffer_enqueue_write(
81  buf1, cq1, CL_FALSE, 0, buf_size, hbuf, NULL, &err);
82  g_assert_no_error(err);
83 
84  /* Transfer data from one buffer to another. */
85  evt = ccl_buffer_enqueue_copy(buf1, buf2, cq2, 0, 0, buf_size,
86  ccl_ewl(&ewl, evt, NULL), &err);
87  g_assert_no_error(err);
88 
89  /* Wait for copy. */
90  ccl_event_wait(ccl_ewl(&ewl, evt, NULL), &err);
91  g_assert_no_error(err);
92 
93  /* Stop profile object timer. */
94  ccl_prof_stop(prof);
95 
96  /* Add both queues to profile object. */
97  ccl_prof_add_queue(prof, "A Queue", cq1);
98  ccl_prof_add_queue(prof, "Another Queue", cq2);
99 
100  /* Process queues. */
101  ccl_prof_calc(prof, &err);
102  g_assert_no_error(err);
103 
104  /* Request some profiling information. */
105  time_ellapsed = ccl_prof_time_elapsed(prof);
106  duration = ccl_prof_get_duration(prof);
107  eff_duration = ccl_prof_get_eff_duration(prof);
108 
109  g_debug("Profilling time ellapsed: %lf", time_ellapsed);
110  g_debug("Profilling duration: %d", (cl_int) duration);
111  g_debug("Profilling eff. duration: %d", (cl_int) eff_duration);
112 
113  /* Destroy buffers. */
114  ccl_buffer_destroy(buf1);
115  ccl_buffer_destroy(buf2);
116 
117  /* Unref cq1, which should not be destroyed because it is held
118  * by the profile object. */
119  ccl_queue_destroy(cq1);
120 
121  /* Destroy the profile object, which will also destroy cq1. cq2
122  * will me merely unrefed and must still be explicitly destroyed. */
123  ccl_prof_destroy(prof);
124 
125  /* Destroy cq2. */
126  ccl_queue_destroy(cq2);
127 
128  /* Destroy the context. */
129  ccl_context_destroy(ctx);
130 
131  /* Confirm that memory allocated by wrappers has been properly
132  * freed. */
133  g_assert(ccl_wrapper_memcheck());
134 
135 }
