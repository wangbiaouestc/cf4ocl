424 static void event_wait_lists_test() {
425 
426  /* Test variables. */
427  CCLContext* ctx = NULL;
428  CCLDevice* dev = NULL;
429  CCLQueue* cq1 = NULL;
430  CCLQueue* cq2 = NULL;
431  CCLBuffer* buf = NULL;
432  CCLEvent* evt = NULL;
433  GError* err = NULL;
434  cl_float host_buf1[8] = { 2.0, 3.5, 4.2, 5.0, 2.2, 199.0, -12.9, -0.01 };
435  cl_float host_buf2[8];
436  CCLEvent* evt_array[2] = { NULL, NULL };
437  CCLEventWaitList ewl = NULL;
438  const cl_event* clevent_ptr;
439  cl_uint num_evts;
440 
441  /* Get the test context with the pre-defined device. */
442  ctx = ccl_test_context_new(&err);
443  g_assert_no_error(err);
444 
445  /* Get first device in context. */
446  dev = ccl_context_get_device(ctx, 0, &err);
447  g_assert_no_error(err);
448 
449  /* Create command queues. */
450  cq1 = ccl_queue_new(ctx, dev, 0, &err);
451  g_assert_no_error(err);
452 
453  cq2 = ccl_queue_new(ctx, dev, 0, &err);
454  g_assert_no_error(err);
455 
456  /* Create a device buffer. */
457  buf = ccl_buffer_new(
458  ctx, CL_MEM_READ_WRITE, 8 * sizeof(cl_float), NULL, &err);
459  g_assert_no_error(err);
460 
461  /* Write something to buffer using command queue 1, get an event. */
462  evt = ccl_buffer_enqueue_write(buf, cq1, CL_FALSE, 0,
463  8 * sizeof(cl_float), host_buf1, NULL, &err);
464  g_assert_no_error(err);
465 
466  /* Read something from buffer using command queue 2, depending on
467  * previous event. */
468  evt = ccl_buffer_enqueue_read(buf, cq2, CL_FALSE, 0,
469  8 * sizeof(cl_float), host_buf2, ccl_ewl(&ewl, evt, NULL), &err);
470  g_assert_no_error(err);
471 
472  /* Wait for read event using ccl_event_wait_list_add_v(). */
473  evt_array[0] = evt;
474  ccl_event_wait_list_add_v(&ewl, evt_array);
475 
476  /* Analise event wait list. We use the bogus function ewl_test_aux()
477  * to avoid the following GCC warning: "the comparison will always
478  * evaluate as true". We know that it will, but this is only
479  * a test. */
480  num_evts = ccl_event_wait_list_get_num_events(ewl_test_aux(&ewl));
481  g_assert_cmpuint(num_evts, ==, 1);
482  clevent_ptr = ccl_event_wait_list_get_clevents(ewl_test_aux(&ewl));
483  g_assert(*clevent_ptr == ccl_event_unwrap(evt));
484 
485  /* Wait on last event. */
486  ccl_event_wait(&ewl, &err);
487  g_assert_no_error(err);
488 
489  /* Check that ewl is NULL. */
490  g_assert(ewl == NULL);
491 
492  /* Check if host buffers contain the same information. */
493  for (cl_uint i = 0; i < 8; ++i)
494  g_assert_cmpfloat(host_buf1[i], ==, host_buf2[i]);
495 
496  /* Re-add completed event to wait list, clear list explicitly. */
497  ccl_event_wait_list_add(&ewl, evt, NULL);
498  g_assert(ewl != NULL);
499  ccl_event_wait_list_clear(&ewl);
500  g_assert(ewl == NULL);
501 
502  /* Clear it again, should throw no error. */
503  ccl_event_wait_list_clear(&ewl);
504  g_assert(ewl == NULL);
505 
506  /* Release wrappers. */
507  ccl_buffer_destroy(buf);
508  ccl_queue_destroy(cq1);
509  ccl_queue_destroy(cq2);
510  ccl_context_destroy(ctx);
511 
512  /* Confirm that memory allocated by wrappers has been properly
513  * freed. */
514  g_assert(ccl_wrapper_memcheck());
515 
516 }
