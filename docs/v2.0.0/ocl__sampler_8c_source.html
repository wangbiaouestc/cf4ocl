30 CL_API_ENTRY cl_sampler CL_API_CALL
31 clCreateSamplerWithProperties(cl_context context,
32  const cl_sampler_properties* sampler_properties,
33  cl_int* errcode_ret) {
34 
35  /* Allocate sampler. */
36  cl_sampler sampler = g_slice_new(struct _cl_sampler);
37  sampler->ref_count = 1;
38 
39  /* Set context. */
40  sampler->context = context;
41  /* Set defaults (only in OpenCL 2.0). */
42  sampler->normalized_coords = CL_TRUE;
43  sampler->addressing_mode = CL_ADDRESS_CLAMP;
44  sampler->filter_mode = CL_FILTER_NEAREST;
45  /* Set defined properties: */
46  if (sampler_properties != NULL) {
47  cl_sampler_properties sp_key;
48  cl_sampler_properties sp_value;
49  for (guint i = 0; (sp_key = sampler_properties[i]) != 0; ++i) {
50  i++;
51  sp_value = sampler_properties[i];
52  switch (sp_key) {
53  case CL_SAMPLER_NORMALIZED_COORDS:
54  sampler->normalized_coords = (cl_bool) sp_value; break;
55  case CL_SAMPLER_ADDRESSING_MODE:
56  sampler->addressing_mode = (cl_addressing_mode) sp_value; break;
57  case CL_SAMPLER_FILTER_MODE:
58  sampler->filter_mode = (cl_filter_mode) sp_value; break;
59  default:
60  seterrcode(errcode_ret, CL_INVALID_VALUE);
61  clReleaseSampler(sampler);
62  return NULL;
63  /* Not checking if sp_value is valid property and not
64  * checking if same sp_key is given more than once. */
65  }
66  }
67  }
68 
69  /* All good. */
70  seterrcode(errcode_ret, CL_SUCCESS);
71  return sampler;
72 }
