37 static void context_with_image_support_setup(
38  CCLContext** ctx_fixt, gconstpointer user_data) {
39 
40  CCLPlatforms* ps;
41  CCLPlatform* p;
42  CCLDevice* d;
43  GError* err = NULL;
44  cl_uint num_devs;
45  *ctx_fixt = NULL;
46  cl_uint min_ocl_ver = 0;
47  if (user_data != NULL)
48  min_ocl_ver = *((cl_uint*) user_data);
49 
50  /* Get all OpenCL platforms in system. */
51  ps = ccl_platforms_new(&err);
52  g_assert_no_error(err);
53 
54  /* Cycle through platforms. */
55  for (guint i = 0; i < ccl_platforms_count(ps); ++i) {
56 
57  /* Get next platform. */
58  p = ccl_platforms_get(ps, i);
59 
60  /* Check if a minimum OpenCL version was set. */
61  if (min_ocl_ver > 0) {
62  /* If so make sure current platform has the required
63  * OpenCL version. */
64  cl_uint p_ocl_ver = ccl_platform_get_opencl_version(p, &err);
65  g_assert_no_error(err);
66  if (p_ocl_ver < min_ocl_ver) {
67  /* If not, go to next platform. */
68  continue;
69  }
70  }
71 
72  /* Get number of devices in current platform. */
73  num_devs = ccl_platform_get_num_devices(p, &err);
74  g_assert_no_error(err);
75 
76  /* Cycle through devices. */
77  for (guint j = 0; j < num_devs; ++j) {
78 
79  /* Get next device. */
80  d = ccl_platform_get_device(p, j, &err);
81  g_assert_no_error(err);
82  /* Check if current device supports images. */
83  cl_bool image_support = ccl_device_get_info_scalar(
84  d, CL_DEVICE_IMAGE_SUPPORT, cl_bool, &err);
85  g_assert_no_error(err);
86  if (image_support) {
87  /* If so, create a context with current device and
88  * return. */
89  *ctx_fixt = ccl_context_new_from_devices(
90  1, (CCLDevice* const*) &d, &err);
91  g_assert_no_error(err);
92  ccl_platforms_destroy(ps);
93  return;
94  }
95  }
96  }
97  ccl_platforms_destroy(ps);
98 }
