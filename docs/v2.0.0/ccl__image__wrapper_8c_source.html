389 CCL_EXPORT
390 CCLImage* ccl_image_new(CCLContext* ctx, cl_mem_flags flags,
391  const cl_image_format* image_format, void* host_ptr, GError** err,
392  ...) {
393 
394  /* Make sure ctx is not NULL. */
395  g_return_val_if_fail(ctx != NULL, NULL);
396  /* Make sure err is NULL or it is not set. */
397  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
398 
399  /* The va_list, which represents the variable argument list. */
400  va_list args_va;
401  /* A key. */
402  const char* key;
403  /* The image description object, initialized to zeros. */
404  CCLImageDesc image_dsc = CCL_IMAGE_DESC_BLANK;
405 
406  /* Initialize the va_list. */
407  va_start(args_va, err);
408 
409  /* Check if any arguments are given, and if so, populate
410  * image_dsc. */
411  while ((key = va_arg(args_va, const char*)) != NULL) {
412 
413  if (g_strcmp0(key, "image_type") == 0) {
414  image_dsc.image_type = va_arg(args_va, cl_mem_object_type);
415  } else if (g_strcmp0(key, "image_width") == 0) {
416  image_dsc.image_width = va_arg(args_va, size_t);
417  } else if (g_strcmp0(key, "image_height") == 0) {
418  image_dsc.image_height = va_arg(args_va, size_t);
419  } else if (g_strcmp0(key, "image_depth") == 0) {
420  image_dsc.image_depth = va_arg(args_va, size_t);
421  } else if (g_strcmp0(key, "image_array_size") == 0) {
422  image_dsc.image_array_size = va_arg(args_va, size_t);
423  } else if (g_strcmp0(key, "image_row_pitch") == 0) {
424  image_dsc.image_row_pitch = va_arg(args_va, size_t);
425  } else if (g_strcmp0(key, "image_slice_pitch") == 0) {
426  image_dsc.image_slice_pitch = va_arg(args_va, size_t);
427  } else if (g_strcmp0(key, "num_mip_levels") == 0) {
428  image_dsc.num_mip_levels = va_arg(args_va, cl_uint);
429  } else if (g_strcmp0(key, "num_samples") == 0) {
430  image_dsc.num_samples = va_arg(args_va, cl_uint);
431  } else if (g_strcmp0(key, "memobj") == 0) {
432  image_dsc.memobj = va_arg(args_va, CCLMemObj*);
433  } else {
434  g_set_error(err, CCL_ERROR, CCL_ERROR_ARGS,
435  "%s: unknown key '%s'", CCL_STRD, key);
436  va_end(args_va);
437  return NULL;
438  }
439  }
440 
441  /* Close the va_list. */
442  va_end(args_va);
443 
444  /* Create the image using the vector version of this function. */
445  return ccl_image_new_v(ctx, flags, image_format,
446  (const CCLImageDesc*) &image_dsc, host_ptr, err);
447 
448 }
