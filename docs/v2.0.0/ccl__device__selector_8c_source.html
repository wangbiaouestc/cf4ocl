572 CCLDevSelDevices ccl_devsel_select(
573  CCLDevSelFilters* filters, GError **err) {
574 
575  /* Make sure filters is not NULL. */
576  g_return_val_if_fail(filters != NULL, NULL);
577 
578  /* Make sure err is NULL or it is not set. */
579  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
580 
581  /* Internal error handling object. */
582  GError* err_internal = NULL;
583 
584  /* Array of devices. */
585  CCLDevSelDevices devices;
586 
587  /* Get all devices present in the system. */
588  devices = ccl_devsel_get_devices(&err_internal);
589  ccl_if_err_propagate_goto(err, err_internal, error_handler);
590 
591  /* *** Filter devices. *** */
592 
593  /* Cycle through all filters. */
594  for (guint i = 0; i < (*filters)->len; i++) {
595 
596  /* If there are no more devices left, get out... */
597  if (devices->len == 0) break;
598 
599  /* Get current filter. */
600  CCLDevSelFilter* curr_filter = g_ptr_array_index(*filters, i);
601 
602  /* Check type of filter, proceed accordingly. */
603  if (curr_filter->type == CCL_DEVSEL_DEP) {
604 
605  /* It's a dependent filter.*/
606  devices = ((ccl_devsel_dep) (curr_filter->function))(
607  devices, curr_filter->data, &err_internal);
608  ccl_if_err_propagate_goto(err, err_internal, error_handler);
609 
610  } else {
611  /* It's an independent filter. */
612 
613  /* Cycle through all devices. */
614  for (guint j = 0; j < devices->len; j++) {
615 
616  /* Get current device wrapper. */
617  CCLDevice* curr_device =
618  (CCLDevice*) g_ptr_array_index(devices, j);
619 
620  /* Check if current device is accepted by current
621  * filter. */
622  cl_bool pass =
623  ((ccl_devsel_indep) curr_filter->function)(
624  curr_device, curr_filter->data, &err_internal);
625  ccl_if_err_propagate_goto(
626  err, err_internal, error_handler);
627 
628  /* If current device didn't pass current filter... */
629  if (!pass) {
630 
631  /* Remove device wrapper from device wrapper array. */
632  g_ptr_array_remove_index(devices, j);
633 
634  /* Force device index decrement, because next device to
635  * be checked will be in this index. */
636  j--;
637 
638  }
639  }
640  }
641  }
642 
643  /* If we got here, everything is OK. */
644  g_assert(err == NULL || *err == NULL);
645  goto finish;
646 
647 error_handler:
648  /* If we got here there was an error, verify that it is so. */
649  g_assert(err == NULL || *err != NULL);
650 
651 finish:
652 
653  /* Free individual filters. */
654  for (guint i = 0; i < (*filters)->len; i++)
655  g_slice_free(CCLDevSelFilter, g_ptr_array_index(*filters, i));
656 
657  /* Free filter array. */
658  g_ptr_array_free(*filters, CL_TRUE);
659 
660  /* Set filters to NULL, so variable can be reused by client. */
661  *filters = NULL;
662 
663  /* Return the selected devices. */
664  return devices;
665 
666 }

815  CCLDevice* dev, void *data, GError **err) {

824 
825  /* Return value, i.e., flag indicating if device belongs to the
826  * specified platform. */
827  cl_bool pass = CL_FALSE;
828 
829  /* Partial name must be a substring of complete name. */
830  gchar *complt_info, *complt_info_lowr, *part_info;
831 
832  /* Make sure data is not NULL. */
833  ccl_if_err_create_goto(*err, CCL_ERROR, data == NULL,
834  CCL_ERROR_INVALID_DATA, error_handler,
835  "%s: invalid filter data", CCL_STRD);
836 
837  /* Lower-case partial name for comparison. */
838  part_info = g_ascii_strdown((gchar*) data, -1);
839 
840  /* Compare with device name. */
841  complt_info = ccl_device_get_info_array(
842  dev, CL_DEVICE_NAME, char*, &err_internal);
843  ccl_if_err_propagate_goto(err, err_internal, error_handler);
844 
845  complt_info_lowr = g_ascii_strdown(complt_info, -1);
846 
847  if (!g_strrstr(complt_info_lowr, part_info)) {
848  /* Device name does not match, check device vendor. */
849 
850  /* Free string allocated for previous comparison. */
851  g_free(complt_info_lowr);
852 
853  /* Compare with device vendor. */
854  complt_info = ccl_device_get_info_array(
855  dev, CL_DEVICE_VENDOR, char*, &err_internal);
856  ccl_if_err_propagate_goto(err, err_internal, error_handler);
857 
858  complt_info_lowr = g_ascii_strdown(complt_info, -1);
859 
860  if (!g_strrstr(complt_info_lowr, part_info)) {
861  /* Device vendor does not match, check platform name. */
862 
863  /* Free string allocated for previous comparison. */
864  g_free(complt_info_lowr);
865 
866  /* Compare with platform name. */

876  ccl_if_err_propagate_goto(err, err_internal, error_handler);
877 
878  complt_info_lowr = g_ascii_strdown(complt_info, -1);
879

882  pass = CL_TRUE;
883  }

904  /* If we got here there was an error, verify that it is so. */
