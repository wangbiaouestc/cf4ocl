66 CCL_EXPORT
67 CCLPlatforms* ccl_platforms_new(GError **err) {
68 
69  /* Make sure err is NULL or it is not set. */
70  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
71 
72  /* Return status of OpenCL functions. */
73  cl_int ocl_status;
74 
75  /* Object which represents the list of OpenCL platforms available
76  * in the system. */
77  CCLPlatforms* platforms = NULL;
78 
79  /* Size in bytes of array of platform IDs. */
80  gsize platf_ids_size;
81 
82  /* Array of platform IDs. */
83  cl_platform_id* platf_ids = NULL;
84 
85  /* Allocate memory for the CCLPlatforms object. */
86  platforms = g_slice_new0(CCLPlatforms);
87 
88  /* Get number of platforms */
89  ocl_status = clGetPlatformIDs(0, NULL, &platforms->num_platfs);
90  ccl_if_err_create_goto(*err, CCL_ERROR,
91  platforms->num_platfs == 0, CCL_ERROR_DEVICE_NOT_FOUND,
92  error_handler, "%s: no OpenCL platforms found.", CCL_STRD);
93  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
94  CL_SUCCESS != ocl_status, ocl_status, error_handler,
95  "%s: get number of platforms (OpenCL error %d: %s).",
96  CCL_STRD, ocl_status, ccl_err(ocl_status));
97 
98  /* Determine size in bytes of array of platform IDs. */
99  platf_ids_size = sizeof(cl_platform_id) * platforms->num_platfs;
100 
101  /* Allocate memory for array of platform IDs. */
102  platf_ids = (cl_platform_id*) g_slice_alloc(platf_ids_size);
103 
104  /* Get existing platform IDs. */
105  ocl_status = clGetPlatformIDs(
106  platforms->num_platfs, platf_ids, NULL);
107  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
108  CL_SUCCESS != ocl_status, ocl_status, error_handler,
109  "%s: get platforms IDs (OpenCL error %d: %s).",
110  CCL_STRD, ocl_status, ccl_err(ocl_status));
111 
112  /* Allocate memory for array of platform wrapper objects. */
113  platforms->platfs =
114  g_slice_alloc(sizeof(CCLPlatform*) * platforms->num_platfs);
115 
116  /* Wrap platform IDs in platform wrapper objects. */
117  for (guint i = 0; i < platforms->num_platfs; i++) {
118  /* Add platform wrapper object to array of wrapper objects. */
119  platforms->platfs[i] = ccl_platform_new_wrap(platf_ids[i]);
120  }
121 
122  /* Free array of platform ids. */
123  g_slice_free1(platf_ids_size, platf_ids);
124 
125  /* If we got here, everything is OK. */
126  g_assert(err == NULL || *err == NULL);
127  goto finish;
128 
129 error_handler:
130 
131  /* If we got here there was an error, verify that it is so. */
132  g_assert(err == NULL || *err != NULL);
133 
134  /* Destroy the CCLPlatforms object, or what was possible to build
135  * of it. */
136  if (platforms) {
137  ccl_platforms_destroy(platforms);
138  }
139 
140  /* Set platforms to NULL, indicating an error occurred.*/
141  platforms = NULL;
142 
143 finish:
144 
145  /* Return the CCLPlatforms object. */
146  return platforms;
147 
148 }

158 CCL_EXPORT
159 void ccl_platforms_destroy(CCLPlatforms* platforms) {
160 
161  /* Platforms object can't be NULL. */
162  g_return_if_fail(platforms != NULL);
163 
164  /* Destroy underlying platforms. */
165  for (guint i = 0; i < platforms->num_platfs; i++) {
166  ccl_platform_unref(platforms->platfs[i]);
167  }
168 
169  /* Free underlying platforms array. */
170  g_slice_free1(
171  sizeof(CCLPlatform*) * platforms->num_platfs, platforms->platfs);
172 
173  /* Free CCLPlatforms object. */
174  g_slice_free(CCLPlatforms, platforms);
175 }
