248 CCL_EXPORT
249 void ccl_kernel_set_args(CCLKernel* krnl, ...) {
250 
251  /* The va_list, which represents the variable argument list. */
252  va_list args_va;
253  /* Array of arguments, created from the va_list. */
254  void** args_array = NULL;
255  /* Number of arguments. */
256  guint num_args = 0;
257  /* Aux. arg. when cycling through the va_list. */
258  void* aux_arg;
259 
260  /* Initialize the va_list. */
261  va_start(args_va, krnl);
262 
263  /* Get first argument. */
264  aux_arg = va_arg(args_va, void*);
265 
266  /* Check if any arguments are given, and if so, populate array
267  * of arguments. */
268  if (aux_arg != NULL) {
269 
270  /* 1. Determine number of arguments. */
271 
272  while (aux_arg != NULL) {
273  num_args++;
274  aux_arg = va_arg(args_va, void*);
275  }
276  va_end(args_va);
277 
278  /* 2. Populate array of arguments. */
279 
280  args_array = g_slice_alloc((num_args + 1) * sizeof(void*));
281  va_start(args_va, krnl);
282 
283  for (guint i = 0; i < num_args; ++i) {
284  aux_arg = va_arg(args_va, void*);
285  args_array[i] = aux_arg;
286  }
287  va_end(args_va);
288  args_array[num_args] = NULL;
289 
290  }
291 
292  /* If any arguments are given... */
293  if (num_args > 0) {
294 
295  /* Call the array version of this function.*/
296  ccl_kernel_set_args_v(krnl, args_array);
297 
298  /* Free the array of arguments. */
299  g_slice_free1((num_args + 1) * sizeof(void*), args_array);
300 
301  }

388 CCL_EXPORT
389 CCLEvent* ccl_kernel_enqueue_ndrange(CCLKernel* krnl, CCLQueue* cq,
390  cl_uint work_dim, const size_t* global_work_offset,
391  const size_t* global_work_size, const size_t* local_work_size,
392  CCLEventWaitList* evt_wait_lst, GError** err) {
393 
394  /* Make sure krnl is not NULL. */
395  g_return_val_if_fail(krnl != NULL, NULL);
396  /* Make sure cq is not NULL. */
397  g_return_val_if_fail(cq != NULL, NULL);
398  /* Make sure err is NULL or it is not set. */
399  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
400 
401  /* OpenCL status flag. */
402  cl_int ocl_status;
403 
404  /* OpenCL event. */
405  cl_event event;
406  /* Event wrapper. */
407  CCLEvent* evt;
408 
409  /* Iterator for table of kernel arguments. */
410  GHashTableIter iter;
411  gpointer arg_index_ptr, arg_ptr;
412 
413  /* Set pending kernel arguments. */
414  if (krnl->args != NULL) {
415  g_hash_table_iter_init(&iter, krnl->args);
416  while (g_hash_table_iter_next(&iter, &arg_index_ptr, &arg_ptr)) {
417  cl_uint arg_index = GPOINTER_TO_UINT(arg_index_ptr);
418  CCLArg* arg = (CCLArg*) arg_ptr;
419  ocl_status = clSetKernelArg(ccl_kernel_unwrap(krnl), arg_index,
420  ccl_arg_size(arg), ccl_arg_value(arg));
421  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
422  CL_SUCCESS != ocl_status, ocl_status, error_handler,
423  "%s: unable to set kernel arg %d (OpenCL error %d: %s).",
424  CCL_STRD, arg_index, ocl_status, ccl_err(ocl_status));
425  g_hash_table_iter_remove(&iter);
426  }
427  }
428 
429  /* Run kernel. */
430  ocl_status = clEnqueueNDRangeKernel(ccl_queue_unwrap(cq),
431  ccl_kernel_unwrap(krnl), work_dim, global_work_offset,
432  global_work_size, local_work_size,
433  ccl_event_wait_list_get_num_events(evt_wait_lst),
434  ccl_event_wait_list_get_clevents(evt_wait_lst), &event);
435  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
436  CL_SUCCESS != ocl_status, ocl_status, error_handler,
437  "%s: unable to enqueue kernel (OpenCL error %d: %s).",
438  CCL_STRD, ocl_status, ccl_err(ocl_status));
439 
440  /* Wrap event and associate it with the respective command queue.
441  * The event object will be released automatically when the command
442  * queue is released. */
443  evt = ccl_queue_produce_event(cq, event);
444 
445  /* Clear event wait list. */
446  ccl_event_wait_list_clear(evt_wait_lst);
447 
448  /* If we got here, everything is OK. */
449  g_assert(err == NULL || *err == NULL);
450  goto finish;
451 
452 error_handler:
453 
454  /* If we got here there was an error, verify that it is so. */
455  g_assert(err == NULL || *err != NULL);
456 
457  /* An error occurred, return NULL to signal it. */
458  evt = NULL;
459 
460 finish:
461 
462  /* Return evt. */
463  return evt;
464 
465 }

869 cl_bool ccl_kernel_suggest_worksizes(CCLKernel* krnl, CCLDevice* dev,
870  cl_uint dims, const size_t* real_worksize, size_t* gws, size_t* lws,
871  GError** err) {
872 
873  /* Make sure dev is not NULL. */
874  g_return_val_if_fail(dev != NULL, CL_FALSE);
875  /* Make sure dims not zero. */
876  g_return_val_if_fail(dims > 0, CL_FALSE);
877  /* Make sure real_worksize is not NULL. */
878  g_return_val_if_fail(real_worksize != NULL, CL_FALSE);
879  /* Make sure lws is not NULL. */
880  g_return_val_if_fail(lws != NULL, CL_FALSE);
881  /* Make sure err is NULL or it is not set. */
882  g_return_val_if_fail(err == NULL || *err == NULL, CL_FALSE);
883 
884  /* The preferred workgroup size. */
885  size_t wg_size_mult = 0;
886  size_t wg_size_max;
887  size_t wg_size = 1, wg_size_aux;
888  size_t* max_wi_sizes;
889  cl_uint dev_dims;
890  cl_bool ret_status;
891  size_t real_ws = 1;
892 
893  /* Error handling object. */
894  GError* err_internal = NULL;
895 
896  /* Check if device supports the requested dims. */
897  dev_dims = ccl_device_get_info_scalar(
898  dev, CL_DEVICE_MAX_WORK_ITEM_DIMENSIONS, cl_uint, &err_internal);
899  ccl_if_err_propagate_goto(err, err_internal, error_handler);
900  ccl_if_err_create_goto(*err, CCL_ERROR, dims > dev_dims,
901  CCL_ERROR_UNSUPPORTED_OCL, error_handler,
902  "%s: device only supports a max. of %d dimension, but %d were requested.",
903  CCL_STRD, dev_dims, dims);
904 
905  /* Get max. work item sizes for device. */
906  max_wi_sizes = ccl_device_get_info_array(
907  dev, CL_DEVICE_MAX_WORK_ITEM_SIZES, size_t*, &err_internal);
908  ccl_if_err_propagate_goto(err, err_internal, error_handler);
909 
910  /* For each dimension, if the user specified a maximum local work
911  * size, the effective maximum local work size will be the minimum
912  * between the user value and the device value. */
913  for (cl_uint i = 0; i < dims; ++i) {
914  if (lws[i] != 0)
915  max_wi_sizes[i] = MIN(max_wi_sizes[i], lws[i]);
916  }
917 
918  /* If kernel is not NULL, query it about workgroup size preferences
919  * and capabilities. */
920  if (krnl != NULL) {
921  /* Determine maximum workgroup size. */
922  wg_size_max = ccl_kernel_get_workgroup_info_scalar(krnl, dev,
923  CL_KERNEL_WORK_GROUP_SIZE, size_t, &err_internal);
924  ccl_if_err_propagate_goto(err, err_internal, error_handler);
925 
926 #ifdef CL_VERSION_1_1
927  /* Determine preferred workgroup size multiple (OpenCL >= 1.1). */
928  cl_uint ocl_ver = ccl_kernel_get_opencl_version(krnl, &err_internal);
929  ccl_if_err_propagate_goto(err, err_internal, error_handler);
930  if (ocl_ver >= 110) {
931  wg_size_mult = ccl_kernel_get_workgroup_info_scalar(
932  krnl, dev, CL_KERNEL_PREFERRED_WORK_GROUP_SIZE_MULTIPLE,
933  size_t, &err_internal);
934  } else {
935  wg_size_mult = wg_size_max;
936  }
937 #else
938  wg_size_mult = wg_size_max;
939 #endif
940  } else {
941 
942  /* Kernel is NULL, use values obtained from device. */
943  wg_size_max = ccl_device_get_info_scalar(
944  dev, CL_DEVICE_MAX_WORK_GROUP_SIZE, size_t, &err_internal);
945  ccl_if_err_propagate_goto(err, err_internal, error_handler);
946  wg_size_mult = wg_size_max;
947  }
948 
949  /* Try to find an appropriate local worksize. */
950  for (cl_uint i = 0; i < dims; ++i) {
951  /* Each lws component is at most the preferred workgroup
952  * multiple or the maximum size of that component in device. */
953  lws[i] = MIN(wg_size_mult, max_wi_sizes[i]);
954  /* Update total workgroup size. */
955  wg_size *= lws[i];
956  /* Update total real worksize. */
957  real_ws *= real_worksize[i];
958  }
959 
960  /* Don't let each component of the local worksize to be
961  * higher than the respective component of the real
962  * worksize. */
963  for (cl_uint i = 0; i < dims; ++i) {
964  while (lws[i] > real_worksize[i]) {
965  lws[i] /= 2;
966  wg_size /= 2;
967  }
968  }
969 
970  /* The total workgroup size can't be higher than the maximum
971  * supported by the device. */
972  while (wg_size > wg_size_max) {
973  wg_size_aux = wg_size;
974  for (int i = dims - 1; i >= 0; --i) {
975  if (lws[i] > 1) {
976  /* Local work size can't be smaller than 1. */
977  lws[i] /= 2;
978  wg_size /= 2;
979  }
980  if (wg_size <= wg_size_max) break;
981  }
982  /* Avoid infinite loops and throw error if wg_size didn't
983  * change. */
984  ccl_if_err_create_goto(*err, CCL_ERROR, wg_size == wg_size_aux,
985  CCL_ERROR_OTHER, error_handler,

987  CCL_STRD, (int) wg_size_max);
988  }
989 
990  /* If output variable gws is not NULL... */
991  if (gws != NULL) {
992  /* ...find a global worksize which is a multiple of the local
993  * worksize and is big enough to handle the real worksize. */
994  for (cl_uint i = 0; i < dims; ++i) {
995  gws[i] = ((real_worksize[i] / lws[i])
996  + (((real_worksize[i] % lws[i]) > 0) ? 1 : 0))
997  * lws[i];
998  }
999  } else {
1000  /* ...otherwise check if found local worksizes are divisors of
1001  * the respective real_worksize. If so keep them, otherwise find
1002  * local worksizes which respect the maximum sizes allowed by
1003  * the kernel and the device, and is a dimension-wise divisor of
1004  * the real_worksize. */
1005  cl_bool lws_are_divisors = CL_TRUE;
1006  for (cl_uint i = 0; i < dims; ++i) {
1007  /* Check if lws[i] is divisor of real_worksize[i]. */
1008  if (real_worksize[i] % lws[i] != 0) {
1009  /* Ops... lws[i] is not divisor of real_worksize[i], so
1010  * we'll have to try and find new lws ahead. */
1011  lws_are_divisors = CL_FALSE;
1012  break;
1013  }
1014  }
1015  /* Is lws divisor of real_worksize, dimension-wise? */
1016  if (!lws_are_divisors) {
1017  /* No, so we'll have to find new lws. */
1018  wg_size = 1;
1019  for (cl_uint i = 0; i < dims; ++i) {
1020 
1021  /* For each dimension, try to use the previously
1022  * found lws[i]. */
1023  if ((real_worksize[i] % lws[i] != 0)
1024  || (lws[i] * wg_size > wg_size_max))
1025  {
1026  /* Previoulsy found lws[i] not usable, find
1027  * new one. Must be a divisor of real_worksize[i]
1028  * and respect the kernel and device maximum lws.*/
1029  cl_uint best_lws_i = 1;
1030  for (cl_uint j = 2; j <= real_worksize[i] / 2; ++j) {
1031  /* If current value is higher than the kernel
1032  * and device limits, stop searching and use
1033  * the best one so far. */
1034  if ((wg_size * j > wg_size_max)
1035  || (j > max_wi_sizes[i])) break;
1036  /* Otherwise check if current value is divisor
1037  * of lws[i]. If so, keep it as the best so
1038  * far. */
1039  if (real_worksize[i] % j == 0)
1040  best_lws_i = j;
1041  }
1042  /* Keep the best divisor for current dimension. */
1043  lws[i] = best_lws_i;
1044  }
1045  /* Update absolute workgroup size (all dimensions). */
1046  wg_size *= lws[i];
1047  }
1048  }
1049  }
1050 
1051  /* If we got here, everything is OK. */
1052  g_assert(err == NULL || *err == NULL);
1053  ret_status = CL_TRUE;
1054  goto finish;
1055

1060  ret_status = CL_FALSE;
1061 
1062 finish:
1063 
1064  /* Return status. */
1065  return ret_status;
