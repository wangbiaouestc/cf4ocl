450 static void ref_unref_test() {
451 
452  CCLContext* ctx = NULL;
453  CCLContext* ctx_cmp = NULL;
454  GError* err = NULL;
455  CCLPlatforms* ps = NULL;
456  CCLPlatform* p = NULL;
457  CCLPlatform* p_1 = NULL;
458  CCLPlatform* p_l = NULL;
459  cl_platform_id cl_p_1 = NULL;
460  cl_platform_id cl_p_l = NULL;
461  CCLDevice* d = NULL;
462  CCLDevice* d_1 = NULL;
463  CCLDevice* d_l = NULL;
464  CCLDevice* const* ds;
465  CCLDevSelFilters filters = NULL;
466  cl_uint num_devs;
467  cl_device_type dev_type;
468 
469  /* ********************************************* */
470  /* **** Test context creating from_devices. **** */
471  /* ********************************************* */
472 
473  /* Get all platforms in system. */
474  ps = ccl_platforms_new(&err);
475  g_assert_no_error(err);
476 
477  /* Cycle through platforms available in system. */
478  for (cl_uint i = 0; i < ccl_platforms_count(ps); ++i) {
479 
480  /* Get current platform. */
481  p = ccl_platforms_get(ps, i);
482  g_assert(p != NULL);
483 
484  /* Cycle through devices available in platform. */
485  num_devs = ccl_platform_get_num_devices(p, &err);
486  g_assert_no_error(err);
487  for (cl_uint j = 0; j < num_devs; ++j) {
488 
489  /* Get current device (this increases ref. count of device
490  * in platform wrapper due to lazy initialization). */
491  d = ccl_platform_get_device(p, j, &err);
492  g_assert_no_error(err);
493 
494  /* Create a context using current device (does not increase
495  * ref count of device wrapper because context wrapper
496  * doesn't hold the device wrappers at this stage). */
497  ctx = ccl_context_new_from_devices(1, &d, &err);
498  g_assert_no_error(err);
499 
500  /* Check the reference count of platform, device and context*/
501  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) p), ==, 1);
502  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d), ==, 1);
503  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) ctx), ==, 1);
504 
505  /* Increase ref. count of context, check that ref count is now 2. */
506  ccl_context_ref(ctx);
507  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) ctx), ==, 2);
508 
509  /* Decrease ref. count of context, check that ref count is now 1. */
510  ccl_context_unref(ctx);
511  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) ctx), ==, 1);
512 
513  /* If this is the first device in platform, keep a
514  * reference to it. */
515  if (j == 0) {
516  d_1 = d;
517  ccl_device_ref(d_1);
518  /* Device ref count should be 1 (1 from platform + 1 from d_1). */
519  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d), ==, 2);
520  }
521 
522  /* If this is the last device in platform, keep a
523  * reference to it. */
524  if (j == num_devs - 1) {
525  d_l = d;
526  ccl_device_ref(d_l);
527  /* If first and last device are the same... */
528  if (num_devs == 1) {
529  /* ...then the total ref count of current (last)
530  * device should be 3 (1 from platform + 1 d_1 + 1 d_l). */
531  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d), ==, 3);
532  } else {
533  /* Otherwise it should be 2 (1 from platform + 1 from d_l). */
534  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d), ==, 2);
535  }
536  }
537 
538  /* Destroy current context: does not decrease ref count of
539  * device because of lazy initialization (device was never
540  * requested from context). */
541  ccl_context_destroy(ctx);
542  }
543 
544  /* If first and last device are the same... */
545  if (num_devs == 1) {
546  /* ...then the ref count of this device should be 3 (from first, last and platform). */
547  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d), ==, 3);
548  /* ...and d_1 and d_l should be the same. */
549  g_assert_cmphex(GPOINTER_TO_UINT(d_1), ==, GPOINTER_TO_UINT(d_l));
550  } else {
551  /* Otherwise it should be 2 for each device (from itself and from platform). */
552  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d_1), ==, 2);
553  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d_l), ==, 2);
554  /* ...and d_1 and d_l should be the different. */
555  g_assert_cmphex(GPOINTER_TO_UINT(d_1), !=, GPOINTER_TO_UINT(d_l));
556  }
557 
558  /* Check that the platform wrappers associated with the first
559  * and last devices in platform are the same object and that
560  * its ref. count is 3 (2 from new_wrap + 1 from reference in
561  * platforms set). */
562  cl_p_1 = ccl_device_get_info_scalar(
563  d_1, CL_DEVICE_PLATFORM, cl_platform_id, &err);
564  g_assert_no_error(err);
565  p_1 = ccl_platform_new_wrap(cl_p_1);
566  cl_p_l = ccl_device_get_info_scalar(
567  d_l, CL_DEVICE_PLATFORM, cl_platform_id, &err);
568  g_assert_no_error(err);
569  p_l = ccl_platform_new_wrap(cl_p_l);
570  g_assert_cmphex(GPOINTER_TO_UINT(p_1), ==, GPOINTER_TO_UINT(p_l));
571  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) p_1), ==, 3);
572  ccl_platform_unref(p_1); /* Could use p_l, its the same object. */
573  ccl_platform_unref(p_1); /* Could use p_l, its the same object. */
574 
575  /* Destroy first and last device in platform (which may be the same). */
576  if (d_1 == d_l) {
577  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d_1), ==, 3);
578  } else {
579  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d_1), ==, 2);
580  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d_l), ==, 2);
581  }
582  ccl_device_destroy(d_1);
583  ccl_device_destroy(d_l);
584  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d_1), ==, 1);
585  g_assert_cmpuint(ccl_wrapper_ref_count((CCLWrapper*) d_l), ==, 1);
586 
587  /* At this time there is still a reference to each device held
588  * in the platform wrapper. */
589 
590  /* Create a context with all devices in current platform. The
591  * devices ref. count should be the same because devices were
592  * already requested (thus lazy initialized) from this platform. */
593  ds = ccl_platform_get_all_devices(p, &err);
594  g_assert_no_error(err);
595  ctx = ccl_context_new_from_devices(num_devs, (CCLDevice**) ds, &err);
596  g_assert_no_error(err);
