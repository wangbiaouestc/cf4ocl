63 static CCLWrapperInfo* ccl_platform_get_cldevices(
64  CCLDevContainer* devcon, GError** err) {
65 
66  /* Make sure err is NULL or it is not set. */
67  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
68 
69  /* Make sure devcon is not NULL. */
70  g_return_val_if_fail(devcon != NULL, NULL);
71 
72  CCLWrapperInfo* info = NULL;
73 
74  cl_int ocl_status;
75 
76  /* Determine number of devices. */
77  ocl_status = clGetDeviceIDs(devcon->base.cl_object,
78  CL_DEVICE_TYPE_ALL, 0, NULL, &devcon->num_devices);
79  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
80  CL_SUCCESS != ocl_status, ocl_status, error_handler,
81  "%s: get number of devices (OpenCL error %d: %s).",
82  CCL_STRD, ocl_status, ccl_err(ocl_status));
83 
84  /* Create info object with size in bytes of array of device IDs. */
85  info = ccl_wrapper_info_new(
86  sizeof(cl_device_id) * devcon->num_devices);
87 
88  /* Get existing device IDs. */
89  ocl_status = clGetDeviceIDs(devcon->base.cl_object,
90  CL_DEVICE_TYPE_ALL, devcon->num_devices, info->value, NULL);
91  ccl_if_err_create_goto(*err, CCL_OCL_ERROR,
92  CL_SUCCESS != ocl_status, ocl_status, error_handler,
93  "%s: get device IDs (OpenCL error %d: %s).",
94  CCL_STRD, ocl_status, ccl_err(ocl_status));
95 
96  /* Add device list to info table, so that it will be
97  * automatically released. Because the cl_platform_id object
98  * doesn't have a CL_PLATFORM_DEVICES parameter, we keep this info
99  * referenced has CL_CONTEXT_DEVICES. */
100  ccl_wrapper_add_info(
101  (CCLWrapper*) devcon, CL_CONTEXT_DEVICES, info);
102 
103  /* If we got here, everything is OK. */
104  g_assert(err == NULL || *err == NULL);
105  goto finish;
106 
107 error_handler:
108  /* If we got here there was an error, verify that it is so. */
109  g_assert(err == NULL || *err != NULL);
110 
111  /* Free info if it was created. */
112  if (info != NULL) ccl_wrapper_info_destroy(info);
113 
114 finish:
115 
116  /* Terminate function. */
117  return info;
118 }

168  g_return_val_if_fail(err == NULL || *err == NULL, NULL);
169 
170  /* The OpenCL platform_id object. */
171  cl_platform_id platform_id;
172  /* The platform wrapper to return. */
173  CCLPlatform* platf = NULL;
174  /* Internal error object. */
175  GError* err_internal = NULL;
176 
177  /* Get OpenCL platform_id object from device. */
178  platform_id = ccl_device_get_info_scalar(

180  ccl_if_err_propagate_goto(err, err_internal, error_handler);

191  g_assert(err == NULL || *err != NULL);
192 
193 finish:
194 
195  /* Return the device platform wrapper. */
