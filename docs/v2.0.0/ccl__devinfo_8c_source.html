120  /* If version was requested, output version and exit. */
121  if (version) {

123  exit(0);

126  /* Check if user requested a list of known information parameters. */
127  if (opt_list) {
128 
129  /*Yes, user requested list, present it. */
130 
131  g_fprintf(CCL_DEVINFO_OUT, "\nKnown information parameters:\n\n");
132  for (gint i = 0; i < ccl_devquery_info_map_size; i++) {
133  if (opt_verb) {
134  g_fprintf(CCL_DEVINFO_OUT,
135  "\t%s\n\t\t%s.\n\n",
136  ccl_devquery_info_map[i].param_name,
137  ccl_devquery_info_map[i].description);
138  } else {

142  }
143  }
144  g_fprintf(CCL_DEVINFO_OUT, "\n");
145 
146  } else {
147 
148  /* User didn't request list, proceed as normal query. */
149 
150  /* Get list of platform wrapper objects. */
151  platforms = ccl_platforms_new(&err);
152  ccl_if_err_goto(err, error_handler);
153 
154  /* Cycle through platforms. */
155  for (guint i = 0; i < ccl_platforms_count(platforms); i++) {
156 
157  /* Get out if this platform is not to be queried. */
158  if ((opt_platf != G_MAXUINT) && (i != opt_platf))
159  continue;
160 
161  /* Get current platform. */
162  p = ccl_platforms_get(platforms, i);
163 
164  /* Show platform information. */
165  ccl_devinfo_show_platform_info(p, i);
166 
167  /* Get number of devices. */
168  num_devs = ccl_platform_get_num_devices(p, &err);
169  ccl_if_err_goto(err, error_handler);
170 
171  /* Cycle through devices. */
172  for (guint j = 0; j < num_devs; j++) {
173 
174  /* Get out if this device is not to be queried. */
175  if ((opt_dev != G_MAXUINT) && (j != opt_dev))
176  continue;
177 
178  /* Get current device. */
179  d = ccl_platform_get_device(p, j, &err);
180  ccl_if_err_goto(err, error_handler);
181 
182  /* Get device name. */
183  info_value = ccl_device_get_info(d, CL_DEVICE_NAME, &err);
184  ccl_if_err_goto(err, error_handler);
185 
186  dev_name = (gchar*) info_value->value;
187 
188  /* Show device information. */
189  g_fprintf(CCL_DEVINFO_OUT,
190  "\n [ Device #%d: %s ]\n\n",
191  j, dev_name);
192  if (opt_all)
193  ccl_devinfo_show_device_info_all(d);
194  else if (opt_custom)
195  ccl_devinfo_show_device_info_custom(d);
196  else
197  ccl_devinfo_show_device_info_basic(d);
198 
199  }
200  g_fprintf(CCL_DEVINFO_OUT, "\n");
201  }
202  }
203  /* If we got here, everything is OK. */
204  g_assert(err == NULL);
205  status = CCL_SUCCESS;
206  goto cleanup;
207 
208 error_handler:
209 
210  /* If we got here there was an error, verify that it is so. */
211  g_assert(err != NULL);
212  g_fprintf(stderr, "%s\n", err->message);
213  status = (err->domain == CCL_ERROR) ? err->code : CCL_ERROR_OTHER;
214  g_error_free(err);
215 
216 cleanup:
217 
218  /* Free stuff! */
219  if (platforms) ccl_platforms_destroy(platforms);
220  g_strfreev(opt_custom);
221 
222  /* Confirm that memory allocated by wrappers has been properly
223  * freed. */
224  g_return_val_if_fail(ccl_wrapper_memcheck(), CCL_ERROR_OTHER);
225 
226  /* Return status. */
227  return status;
228

239 void ccl_devinfo_args_parse(int argc, char* argv[], GError** err) {
240 
241  /* Make sure err is NULL or it is not set. */
242  g_return_if_fail(err == NULL || *err == NULL);
243 
244  /* Command line options context. */
245  GOptionContext* context = NULL;
246 
247  /* Create parsing context. */
248  context = g_option_context_new(" - " CCL_DEVINFO_DESCRIPTION);
249 
250  /* Add acceptable command line options to context. */
251  g_option_context_add_main_entries(context, entries, NULL);
252 
253  /* Use context to parse command line options. */
254  g_option_context_parse(context, &argc, &argv, err);
255  ccl_if_err_goto(*err, error_handler);
256 
257  /* If we get here, no need for error treatment, jump to cleanup. */
258  g_assert(*err == NULL);
259  goto cleanup;
260 
261 error_handler:
262 
263  /* If we got here, everything is OK. */
264  g_assert(*err != NULL);
265 
266 cleanup:
267 
268  /* Free context. */
269  if (context) g_option_context_free(context);
270 
271  /* Return. */
272  return;
